125  Bool Overlap(const AxisAlignedBox& Box, const Sphere& Ball)
126  {
127  // Arvo's algorithm
128  Real SquareVar, Dist = 0;
129  for( Whole X = 0 ; X < 3 ; ++X )
130  {
131  if ( Ball.Center[X] < Box.MinExt[X] ) {
132  SquareVar = Ball.Center[X] - Box.MinExt[X];
133  Dist += SquareVar * SquareVar;
134  }else if( Ball.Center[X] > Box.MaxExt[X] ) {
135  SquareVar = Ball.Center[X] - Box.MaxExt[X];
136  Dist += SquareVar * SquareVar;
137  }
138  }
139  return ( Dist <= Ball.Radius * Ball.Radius );
140  }
