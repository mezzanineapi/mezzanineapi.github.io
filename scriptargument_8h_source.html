54  /// @brief The interface for a script argument
55  /// @details All the members that all script for all languages must implement.
56  /// @n @n
57  /// These are created to provide data to scripts. This is intended to primarily
58  /// to act as an abstraction layer between game data and data inside scripts,
59  /// but currently just transports primitives into and out of scripts without needing to care
60  /// about any underlying types. This limits what operations can be done with
61  /// this, because it is intended to provide an abstraction that can be used
62  /// to implement specific algorithms for languages that require them.

113  /// @brief A generic implementation of a ScriptArgument that is suitable for primitive types in most situations
114  template <class T>
115  class MEZZ_LIB ScriptArgumentGeneric : public iScriptArgument
116  {
117  protected:
118  /// @brief The actual data.
119  /// @warning To prevent Slicing don't store the parent class in a container.
120  /// This practically guarantees that derived class will have different sizes
121  /// and overflow container boundaries.
122  T Datum;
123 
124  public:
125  /// @brief To make working with this easier.
126  typedef T Type;
127 
128  /// @brief Create an initialized Argument
129  /// @param InitialValue The value to initialize the Argument with.
130  /// @note Intentionally not explicit, this allow for passing convertable types directly to functions.
131  ScriptArgumentGeneric(T InitialValue) :
132  Datum(InitialValue)
133  {}
134 
135  /// @brief Overloadable Deconstructor
136  virtual ~ScriptArgumentGeneric()
137  {}
138 
139  /// @brief Get the Argument as a String, slow default implementation.
140  /// @return The argument value lexographically converted as a @ref String
141  virtual String GetString() const
142  { return ToString(Datum); }
143 
144  /// @brief Get the Argument as a Whole, slow default implementation.
145  /// @return The argument value lexographically converted as a @ref Whole
146  virtual Whole GetWhole() const
147  { return ToWhole(Datum); }
148 
149  /// @brief Get the Argument as a Integer, slow default implementation.
150  /// @return The argument value lexographically converted as an @ref Integer
151  virtual Integer GetInteger() const
152  { return ToInteger(Datum); }
153 
154  /// @brief Get the Argument as a Real, slow default implementation.
155  /// @return The argument value lexographically converted as an @ref Real
156  virtual Real GetReal() const
157  { return ToReal(Datum); }
158 
159  /// @brief Get the Argument as a Bool
160  /// @return The argument value lexographically converted as an @ref Bool
161  virtual Bool GetBool() const
162  { return ToBool(Datum); }
163 
164  /// @brief Provide an overload point to change assignment that operators will use.
165  /// @param NewValue The new value for this.
166  virtual void SetValue(T NewValue)
167  { Datum=NewValue; }
168 
169  /// @brief Get the raw primitive to value.
170  /// @return The internal value that meaningful operation can be performed on.
171  virtual T GetValue() const
172  { return Datum; }
173 
174  /// @brief Get data about this having an unknown type
175  /// @return This will return an Integer containing GenericUnknown. If needed proper specialization of this can be made for any class, see the ScriptArgumentGeneric<Mezzanine::Scripting::Lua::Lua51Nil>
176  virtual Integer GetTypeData() const
177  { return GenericUnknown; }
178 
179  /// @brief Get a pointer to the most Derived type of this class
180  /// @return A pointer of the most derived pointing to this.
181  virtual ScriptArgumentGeneric<T>* GetMostDerived()
182  { return this; }
