205  int GameWindow::IsLargerThenDesktop(const Whole& Width, const Whole& Height)
206  {

221  {
222  Whole Order = ( ZOrder == 0 ? this->Viewports.size() : ZOrder );

318  void GameWindow::SetRenderResolution(const Whole& Width, const Whole& Height)
319  {
320  if( this->Settings.RenderWidth == Width && this->Settings.RenderHeight == Height )
321  return;
322  if( this->Settings.Fullscreen )
323  {
324  SDL_DisplayMode CurrentDisplay;
325  SDL_GetWindowDisplayMode(SDLWindow,&CurrentDisplay);
326  CurrentDisplay.w = Width;
327  CurrentDisplay.h = Height;
328  //CurrentDisplay.refresh_rate = 60;
329  if(SDL_SetWindowDisplayMode(SDLWindow,&CurrentDisplay) == 0)
330  {
331  this->OgreWindow->setFullscreen(true,Width,Height);
332  this->UpdateViewportsAndCameras();
333  this->Settings.RenderWidth = Width;
334  this->Settings.RenderHeight = Height;
335  return;
336  }
337  }else{
338  int Result = this->IsLargerThenDesktop(Width,Height);
339  if(Result == 0)
340  {
341  Whole ResultWidth, ResultHeight;
342  crossplatform::SanitizeWindowedRes(Width,Height,ResultWidth,ResultHeight);
343  SDL_SetWindowSize(SDLWindow,ResultWidth,ResultHeight);
344  this->OgreWindow->setFullscreen(false,ResultWidth,ResultHeight);
345  }else if(Result == 1){
346  Entresol::GetSingletonPtr()->Log("Cannot create a window larger then the desktop resolution.");
347  return;
348  }else{
349  SDL_SetWindowSize(SDLWindow,Width,Height);
350  this->OgreWindow->setFullscreen(false,Width,Height);
351  }
352  this->UpdateViewportsAndCameras();
353  this->Settings.RenderWidth = Width;
354  this->Settings.RenderHeight = Height;
355  }
356  }

358  void GameWindow::SetFullscreen(const bool Fullscreen)
359  {
360  static SDL_DisplayMode FSDisplayMode;
361 
362  if( Fullscreen == this->Settings.Fullscreen )
363  return;
364 
365  if( !Fullscreen && this->Settings.Fullscreen )
366  {
367  const WindowSettings& DeskSet = this->Manager->GetDesktopSettings();
368  if( this->Settings.RenderWidth > DeskSet.RenderWidth || this->Settings.RenderHeight > DeskSet.RenderHeight )
369  {
370  this->Settings.RenderWidth = DeskSet.RenderWidth;
371  this->Settings.RenderHeight = DeskSet.RenderHeight;
372  }
373  if( this->Settings.RenderWidth == DeskSet.RenderWidth || this->Settings.RenderHeight == DeskSet.RenderHeight )
374  {
375  Whole ResultWidth, ResultHeight;
376  crossplatform::SanitizeWindowedRes(Settings.RenderWidth,Settings.RenderHeight,ResultWidth,ResultHeight);
377  this->SetRenderResolution(ResultWidth,ResultHeight);
378  this->Settings.RenderWidth = DeskSet.RenderWidth;
379  this->Settings.RenderHeight = DeskSet.RenderHeight;
380  }
381  }
382  else if(Fullscreen && !Settings.Fullscreen)
383  {
384  FSDisplayMode.w = this->Settings.RenderWidth;
385  FSDisplayMode.h = this->Settings.RenderHeight;
386  FSDisplayMode.refresh_rate = Settings.RefreshRate;
387  SDL_SetWindowDisplayMode(SDLWindow,&FSDisplayMode);
388  }
389 
390  if(SDL_SetWindowFullscreen(SDLWindow, Fullscreen?SDL_TRUE:SDL_FALSE ) == 0)
391  {
392  this->OgreWindow->setFullscreen(Fullscreen,this->Settings.RenderWidth,this->Settings.RenderHeight);
393  this->UpdateViewportsAndCameras();
394  this->Settings.Fullscreen = Fullscreen;
395  }
396  }
