103  void TextureAtlas::Load(const String& MTAFile, const String& Group)
104  {
105  Mezzanine::Resource::TextSettingsFile MTA;
106  MTA.Load(MTAFile,Group);
107  Mezzanine::Resource::TextSettingsFile::SectionIterator SecIt = MTA.GetSectionBegin();
108  Mezzanine::Resource::TextSettingsFile::SectionIterator SecEnd = MTA.GetSectionEnd();
109  String AtlasName = MTAFile.substr(0,MTAFile.find_first_of("."));
110  String SectionName;
111 
112  SecIt++;
113  while(SecIt != SecEnd)
114  {
115  SectionName = (*SecIt).first;
116  StringTools::ToLowerCase(SectionName);
117  Mezzanine::Resource::TextSettingsFile::SettingsMap* CurrSection = (*SecIt).second;
118 
119  if(SectionName == "texture")
120  LoadTexture(CurrSection);
121  else if(StringTools::StartsWith(SectionName,"font.",false))
122  {
123  UInt32 Index = StringTools::ConvertToUInt32(SectionName.substr(5));
124  GlyphData* GlyphSet = new GlyphData();
125  Glyphs[Index] = GlyphSet;
126 
127  LoadGlyphs(CurrSection, GlyphSet);
128  LoadKerning(CurrSection, GlyphSet);
129  LoadVerticalOffsets(CurrSection, GlyphSet);
130  GlyphSet->Atlas = AtlasName;
131  }
132  else if(SectionName == "sprites")
133  LoadSprites(CurrSection,AtlasName);
134  SecIt++;
135  }
136  }

138  void TextureAtlas::LoadTexture(Resource::TextSettingsFile::SettingsMap* Config)
139  {
140  String Name, Data;
141  Resource::TextSettingsFile::SettingsIterator It;
142  for(It = Config->begin() ; It != Config->end() ; ++It)
143  {
144  Name = It->first;
145  Data = It->second;
146 
147  size_t Comment = Data.find_first_of('#');
148  if(Comment != String::npos)
149  Data = Data.substr(0,Comment);
150 
151  StringTools::ToLowerCase(Name);
152 
153  // file texturename ~groupname
154  // > file myatlas.png
155  // > file myatlas.png ~loadinggroup
156  if(Name == "file")
157  {
158  String TextureName = Data;
159  String GroupName = "UI";
160  size_t GroupSplit = Data.find_first_of('~');
161  if(GroupSplit != String::npos)
162  {
163  TextureName = Data.substr(0,GroupSplit);
164  GroupName = Data.substr(GroupSplit+1);
165  StringTools::Trim(TextureName);
166  StringTools::Trim(GroupName);
167  }
168  this->TAID->TATexture = Ogre::TextureManager::getSingletonPtr()->getByName(Data,GroupName);
169  if(this->TAID->TATexture.isNull())
170  {
171  this->TAID->TATexture = Ogre::TextureManager::getSingletonPtr()->load(TextureName,GroupName,Ogre::TEX_TYPE_2D,0);
172  }
173 
174  InverseTextureSize.X = 1.0 / this->TAID->TATexture->getWidth();
175  InverseTextureSize.Y = 1.0 / this->TAID->TATexture->getHeight();
176 
177  continue;
178  }
179  else if(Name == "whitepixel")
180  {
181  WhitePixel = StringTools::ConvertToVector2(Data);
182  WhitePixel.X *= InverseTextureSize.X;
183  WhitePixel.Y *= InverseTextureSize.Y;
184  }
185  }

188  void TextureAtlas::LoadGlyphs(Resource::TextSettingsFile::SettingsMap* Config, GlyphData* GlyphD)
189  {
190  String Name, Data;
191  Resource::TextSettingsFile::SettingsIterator It;
192 
193  CountedPtr<StringVector> StrValues;
194  Vector2 Offset;
195  for(It = Config->begin() ; It != Config->end() ; ++It)
196  {
197  Name = It->first;
198  Data = It->second;
199 
200  size_t Comment = Data.find_first_of('#');
201  if(Comment != String::npos)
202  Data = Data.substr(0, Comment);
203  StringTools::ToLowerCase(Name);
204 
205  // lineheight x y
206  // > offset 0 80
207  if (Name == "offset")
208  {
209  Offset = StringTools::ConvertToVector2(Data);
210  continue;
211  }
212  // lineheight lineheight
213  // > lineheight 15
214  else if (Name == "lineheight")
215  {
216  GlyphD->LineHeight = StringTools::ConvertToReal(Data);
217  continue;
218  }
219  // spacelength spacelength
220  // > spacelength 3
221  else if (Name == "spacelength")
222  {
223  GlyphD->SpaceLength = StringTools::ConvertToReal(Data);
224  continue;
225  }
226  // spacelength spacelength
227  // > spacelength 3
228  else if (Name == "baseline")
229  {
230  GlyphD->Baseline = StringTools::ConvertToReal(Data);
231  continue;
232  }
233  // monowidth width
234  // > monowidth 3
235  else if (Name == "monowidth")
236  {
237  GlyphD->MonoWidth = StringTools::ConvertToReal(Data);
238  continue;
239  }
240  // range lower upper
241  // > range 33 126
242  else if (Name == "range")
243  {
244  Vector2 Range = StringTools::ConvertToVector2(Data);
245  GlyphD->RangeBegin = (UInt32)Range.X;
246  GlyphD->RangeEnd = (UInt32)Range.Y;
247  }
248  // kerning kerning
249  // > kerning -1
250  else if(Name == "letterspacing")
251  {
252  GlyphD->LetterSpacing = StringTools::ConvertToReal(Data);
253  }
254  }
255 
256  for(UInt32 Index = GlyphD->RangeBegin ; Index <= GlyphD->RangeEnd ; Index++)
257  {
258  Glyph* NewGlyph = new Glyph();
259  GlyphD->Glyphs.push_back(NewGlyph);
260 
261  std::stringstream GlyphStr;
262  GlyphStr << "glyph_" << Index;
263 
264  It = Config->find(GlyphStr.str());
265  if(It == Config->end())
266  continue;
267 
268  StrValues = StringTools::Split((*It).second," ",5);
269 
270  if(StrValues->size() < 4)
271  {
272  continue;
273  }
274 
275  NewGlyph->UVLeft = Offset.X + StringTools::ConvertToReal( StrValues->at(0) );
276  NewGlyph->UVTop = Offset.Y + StringTools::ConvertToReal( StrValues->at(1) );
277  NewGlyph->UVWidth = StringTools::ConvertToReal( StrValues->at(2) );
278  NewGlyph->UVHeight = StringTools::ConvertToReal( StrValues->at(3) );
279  NewGlyph->UVRight = NewGlyph->UVLeft + NewGlyph->UVWidth;
280  NewGlyph->UVBottom = NewGlyph->UVTop + NewGlyph->UVHeight;
281 
282  if(StrValues->size() == 5) NewGlyph->GlyphAdvance = StringTools::ConvertToReal( StrValues->at(4) );
283  else NewGlyph->GlyphAdvance = NewGlyph->UVWidth;
284 
285  NewGlyph->GlyphChar = (GlyphD->Glyphs.size() - 1) + GlyphD->RangeBegin;
286  }
287  }

289  void TextureAtlas::LoadKerning(Resource::TextSettingsFile::SettingsMap* Config, GlyphData* GlyphD)
290  {
291  String LeftName, Data;
292  Resource::TextSettingsFile::SettingsIterator It;
293  UInt32 LeftGlyphID;
294  UInt32 RightGlyphID;
295  Real Kern;
296  CountedPtr<StringVector> StrValues;
297 
298  for(It = Config->begin(); It != Config->end(); ++It)
299  {
300  LeftName = It->first;
301  Data = It->second;
302  StringTools::ToLowerCase(LeftName);
303 
304  if(LeftName.substr(0,6) != "kerning_")
305  continue;
306 
307  size_t Comment = Data.find_first_of('#');
308  if(Comment != String::npos)
309  Data = Data.substr(0,Comment);
310 
311  LeftName = LeftName.substr(8); // chop of kerning_
312  LeftGlyphID = StringTools::ConvertToUInt32(LeftName);
313  StrValues = StringTools::Split(Data," ",2);
314 
315  if(StrValues->size() != 2)
316  {
317  continue;
318  }
319 
320  RightGlyphID = StringTools::ConvertToUInt32(StrValues->at(0));
321  Kern = StringTools::ConvertToReal(StrValues->at(1));
322  GlyphD->Glyphs[RightGlyphID - GlyphD->RangeBegin]->Kernings.push_back(KerningInfo(LeftGlyphID,Kern));
323  }

412  Ogre::MaterialPtr TextureAtlas::GetOrCreate2DMasterMaterial()
413  {
414  Ogre::MaterialPtr Material2D = Ogre::MaterialManager::getSingletonPtr()->getByName("Mezz2D");
415  if(Material2D.isNull() == false)
416  {
417  Ogre::Pass* MatPass = Material2D->getTechnique(0)->getPass(0);
418 
419  if(MatPass->hasVertexProgram())
420  {
421  Ogre::GpuProgramPtr gpuPtr = MatPass->getVertexProgram();
422  gpuPtr->load();
423  }
424 
425  if(MatPass->hasFragmentProgram())
426  {
427  Ogre::GpuProgramPtr gpuPtr = MatPass->getFragmentProgram();
428  gpuPtr->load();
429  }
430 
431  return Material2D;
432  }
433 
434  Material2D = Ogre::MaterialManager::getSingletonPtr()->create("Mezz2D","UI");
435  Ogre::Pass* MatPass = Material2D->getTechnique(0)->getPass(0);
436  MatPass->setCullingMode(Ogre::CULL_NONE);
437  MatPass->setDepthCheckEnabled(false);
438  MatPass->setDepthWriteEnabled(false);
439  MatPass->setLightingEnabled(false);
440  MatPass->setSceneBlending(Ogre::SBT_TRANSPARENT_ALPHA);
441 
442  Ogre::TextureUnitState* TexUnit = MatPass->createTextureUnitState();
443  TexUnit->setTextureAddressingMode(Ogre::TextureUnitState::TAM_CLAMP);
444  TexUnit->setTextureFiltering(Ogre::FO_NONE, Ogre::FO_NONE, Ogre::FO_NONE);
445  }

482  void TextureAtlas::CalculateCoordinates()
483  {
484  Ogre::RenderSystem* rs = Ogre::Root::getSingletonPtr()->getRenderSystem();
485 
486  Real TexelX = rs->getHorizontalTexelOffset();
487  Real TexelY = rs->getVerticalTexelOffset();
488 
489  for(std::map<UInt32, GlyphData*>::iterator gd_it = Glyphs.begin(); gd_it != Glyphs.end(); gd_it++)
490  {
491  for(std::vector<Glyph*>::iterator it = (*gd_it).second->Glyphs.begin(); it != (*gd_it).second->Glyphs.end(); it++)
492  {
493  (*it)->UVLeft -= TexelX;
494  (*it)->UVTop -= TexelY;
495  (*it)->UVRight += TexelX;
496  (*it)->UVBottom += TexelY;
497 
498  (*it)->UVLeft *= InverseTextureSize.X;
499  (*it)->UVTop *= InverseTextureSize.Y;
500  (*it)->UVRight *= InverseTextureSize.X;
501  (*it)->UVBottom *= InverseTextureSize.Y;
502 
503  (*it)->AtlasCoords[QC_TopLeft].X = (*it)->UVLeft;
504  (*it)->AtlasCoords[QC_TopLeft].Y = (*it)->UVTop;
505  (*it)->AtlasCoords[QC_TopRight].X = (*it)->UVRight;
506  (*it)->AtlasCoords[QC_TopRight].Y = (*it)->UVTop;
507  (*it)->AtlasCoords[QC_BottomRight].X = (*it)->UVRight;
508  (*it)->AtlasCoords[QC_BottomRight].Y = (*it)->UVBottom;
509  (*it)->AtlasCoords[QC_BottomLeft].X = (*it)->UVLeft;
510  (*it)->AtlasCoords[QC_BottomLeft].Y = (*it)->UVBottom;
511 
512  (*it)->GlyphSize.X = (*it)->UVWidth;
513  (*it)->GlyphSize.Y = (*it)->UVHeight;
514  }
515  }
516 
517  for(std::map<String, Sprite*>::iterator it = Sprites.begin(); it != Sprites.end(); it++)
518  {
519  (*it).second->UVRight = (*it).second->UVLeft + (*it).second->SpriteSize.X;
520  (*it).second->UVBottom = (*it).second->UVTop + (*it).second->SpriteSize.Y;
521 
522  (*it).second->UVLeft *= InverseTextureSize.X;
523  (*it).second->UVTop *= InverseTextureSize.Y;
524  (*it).second->UVRight *= InverseTextureSize.X;
525  (*it).second->UVBottom *= InverseTextureSize.Y;
526 
527  (*it).second->AtlasCoords[QC_TopLeft].X = (*it).second->UVLeft;
528  (*it).second->AtlasCoords[QC_TopLeft].Y = (*it).second->UVTop;
529  (*it).second->AtlasCoords[QC_TopRight].X = (*it).second->UVRight;
530  (*it).second->AtlasCoords[QC_TopRight].Y = (*it).second->UVTop;
531  (*it).second->AtlasCoords[QC_BottomRight].X = (*it).second->UVRight;
532  (*it).second->AtlasCoords[QC_BottomRight].Y = (*it).second->UVBottom;
533  (*it).second->AtlasCoords[QC_BottomLeft].X = (*it).second->UVLeft;
534  (*it).second->AtlasCoords[QC_BottomLeft].Y = (*it).second->UVBottom;
535  }
536  }
