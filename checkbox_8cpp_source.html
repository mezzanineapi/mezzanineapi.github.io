85  void CheckBox::ConstructCheckbox()
86  {
87  // Add our new events
88  this->AddEvent(CheckBox::EventChecked);
89  this->AddEvent(CheckBox::EventUnchecked);
90  // Add some more render layer groups
91  RenderLayerGroup* CheckedNormalGroup = this->CreateRenderLayerGroup("CheckedNormal");
92  RenderLayerGroup* CheckedHoveredGroup = this->CreateRenderLayerGroup("CheckedHovered");
93 
94  this->StateGroupBindings[ WS_Checked ] = CheckedNormalGroup;
95  this->StateGroupBindings[ WS_Checked | WS_Hovered ] = CheckedHoveredGroup;
96  this->StateGroupBindings[ WS_Checked | WS_Focused ] = CheckedNormalGroup;
97  this->StateGroupBindings[ WS_Checked | WS_Dragged ] = CheckedNormalGroup;
98  this->StateGroupBindings[ WS_Checked | WS_Hovered | WS_Focused ] = CheckedHoveredGroup;
99  this->StateGroupBindings[ WS_Checked | WS_Focused | WS_Dragged ] = CheckedNormalGroup;
100  this->StateGroupBindings[ WS_Checked | WS_Dragged | WS_Hovered ] = CheckedHoveredGroup;
101  this->StateGroupBindings[ WS_Checked | WS_Hovered | WS_Focused | WS_Dragged ] = CheckedHoveredGroup;
102  }

143  void CheckBox::ProtoSerializeProperties(XML::Node& SelfRoot) const
144  {
145  this->Button::ProtoSerializeProperties(SelfRoot);
146 
147  XML::Node PropertiesNode = SelfRoot.AppendChild( CheckBox::GetSerializableName() + "Properties" );
148 
149  if( PropertiesNode.AppendAttribute("Version").SetValue("1") &&
150  PropertiesNode.AppendAttribute("CheckLock").SetValue( this->CheckLock ? "true" : "false" ) )
151  {
152  return;
153  }else{
154  SerializeError("Create XML Attribute Values",CheckBox::GetSerializableName() + "Properties",true);
155  }
156  }
157 
158  void CheckBox::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
159  {
160  this->Button::ProtoDeSerializeProperties(SelfRoot);
161 
162  XML::Attribute CurrAttrib;
163  XML::Node PropertiesNode = SelfRoot.GetChild( CheckBox::GetSerializableName() + "Properties" );
164 
165  if( !PropertiesNode.Empty() ) {
166  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
167  CurrAttrib = PropertiesNode.GetAttribute("LockoutTime");
168  if( !CurrAttrib.Empty() )
169  this->CheckLock = StringTools::ConvertToBool( CurrAttrib.AsString() );
170  }else{
171  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (CheckBox::GetSerializableName() + "Properties") + ": Not Version 1.");
172  }
173  }else{
174  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,CheckBox::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
175  }
176  }

204  void CheckBox::_OnChecked()
205  {
206  this->State |= WS_Checked;
207  this->SetGroupFromState(State);
208 
209  WidgetEventArguments Args(CheckBox::EventChecked,this->Name);
210  this->FireEvent(Args);
211  }

213  void CheckBox::_OnUnchecked()
214  {
215  if( State & WS_Checked )
216  {
217  State &= ~WS_Checked;
218  this->SetGroupFromState(State);
219  }
220 
221  WidgetEventArguments Args(CheckBox::EventUnchecked,this->Name);
222  this->FireEvent(Args);
223  }
