106  {

108  MEZZ_EXCEPTION(Exception::MM_OUT_OF_BOUNDS_EXCEPTION,"Attempting to access invalid index in Collision.");

110  if(Index > 3)

112  Whole SuperIndex = Index/4;
113  Whole SubIndex = Index%4;
114  return (InternalData->Manifolds.at(SuperIndex)->getContactPoint(SubIndex));

116  return (InternalData->Manifolds.at(0)->getContactPoint(Index));

212  void Collision::Update()
213  {
214  InternalAlgo->getAllContactManifolds(InternalData->Manifolds);
215  Whole NumManifolds = InternalData->Manifolds.size();
216  if( PenetrationDistances.size() != NumManifolds )
217  {
218  UpdatePenetrationDistances();
219  ObjectA->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
220  ObjectB->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
221  return;
222  }
223  for( Whole X = 0 ; X < NumManifolds ; ++X )
224  {
225  if( PenetrationDistances[X] != GetDistance(X) )
226  {
227  UpdatePenetrationDistances();
228  ObjectA->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
229  ObjectB->_NotifyCollisionState(this,Collision::Col_Contacts_Updated);
230  return;
231  }
232  }
