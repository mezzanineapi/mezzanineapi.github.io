238  void AreaEffect::_Update()
239  {
240  // Notes for the new post refactor way to update
241  // Everything in vectors. Create a new temp vector composed of the old frames "overlapping" items.
242  // Go through all items detected by ghost proxy and attempt to remove them from the temporary vector.
243  // If it successfully removes, add it to the current overlapping vector. If it fails add to the added vector.
244  // Once one iteration of this is done, all the remaining items in the temp vector are placed in the removed vector.
245 
246  if ( !AddedActors.empty() )
247  this->AddedActors.clear();
248  if ( !RemovedActors.empty() )
249  this->RemovedActors.clear();
250  btSoftRigidDynamicsWorld* PhysWorld = Entresol::GetSingletonPtr()->GetPhysicsManager()->_GetPhysicsWorldPointer();
251 
252  std::list<ActorBase*>::iterator it = OverlappingActors.begin();
253  // Make a bool vector to keep track of which actors to keep when updating.
254  std::vector<bool> Tracker;
255  Tracker.resize(OverlappingActors.size());
256  std::vector<bool>::iterator bit;
257  for ( bit = Tracker.begin() ; bit != Tracker.end() ; bit++ )
258  {
259  (*bit) = false;
260  }
261 
262  btManifoldArray manifoldArray;
263  btBroadphasePairArray& pairArray = Ghost->getOverlappingPairCache()->getOverlappingPairArray();
264  int numPairs = pairArray.size();
265  for (int i=0;i<numPairs;i++)
266  {
267  manifoldArray.clear();
268  const btBroadphasePair& pair = pairArray[i];
269  btBroadphasePair* collisionPair = PhysWorld->getPairCache()->findPair(pair.m_pProxy0,pair.m_pProxy1);
270  if (!collisionPair)
271  continue;
272  if (collisionPair->m_algorithm)
273  collisionPair->m_algorithm->getAllContactManifolds(manifoldArray);
274 
275  for (int j=0;j<manifoldArray.size();j++)
276  {
277  btPersistentManifold* manifold = manifoldArray[j];
278  for (int p=0;p<manifold->getNumContacts();p++)
279  {
280  //const btManifoldPoint& pt = manifold->getContactPoint(p);
281  //if(pt.m_distance1 > 0)
282  // continue;
283  btCollisionObject* ColObj = manifold->getBody0() != Ghost ? (btCollisionObject*)(manifold->getBody0()) : (btCollisionObject*)(manifold->getBody1());
284 
285  WorldObject* WO = static_cast<WorldObject*>(ColObj->getUserPointer());
286  ActorBase* Actor = NULL;
287  if( Mezzanine::WSO_TerrainFirst > WO->GetType() )
288  Actor = static_cast<ActorBase*>( WO );
289  else
290  continue;
291  // Check list for the actor in the pair.
292  for( it = this->OverlappingActors.begin(), bit = Tracker.begin() ; it != this->OverlappingActors.end() ; it++, bit++ )
293  {
294  if ( Actor == (*it) )
295  {
296  (*bit) = true;
297  break;
298  }
299  }
300  if ( it == this->OverlappingActors.end() )
301  {
302  this->AddActorToList(Actor);
303  Tracker.push_back(true);
304  }
305  }
306  }
307  }
308 
309  // Verify they are the same size. Then remove items from the list as necessary.
310  if ( OverlappingActors.size() == Tracker.size() )
311  {
312  std::list<ActorBase*>::iterator sit = OverlappingActors.begin();
313  for ( bit = Tracker.begin() ; bit != Tracker.end() ; )
314  {
315  if ( (*bit) == false )
316  {
317  bit = Tracker.erase(bit);
318  ActorBase* Act = (*sit);
319  RemovedActors.push_back(Act);
320  sit = OverlappingActors.erase(sit);
321  }else{
322  sit++;
323  bit++;
324  }
325  }
326  }// */

479  void GravityWell::ApplyEffect()
480  {
481  if(0 == Strength)
482  return;
483  //ActorBase* Act = NULL;
484  ActorRigid* ActRig = NULL;
485  if(!AllowWorldGrav && !AddedActors.empty())
486  {
487  for ( std::vector<ActorBase*>::iterator AA = AddedActors.begin() ; AA != AddedActors.end() ; AA++ )
488  {
489  if(Mezzanine::WSO_ActorRigid != (*AA)->GetType())
490  continue;
491  ActRig = dynamic_cast<ActorRigid*>(*AA);
492  ActRig->GetPhysicsSettings()->SetIndividualGravity(Vector3());
493  }
494  }
495  if(!OverlappingActors.empty())
496  {
497  Vector3 ActorLoc, Direction;
498  Real Distance, AppliedStrength, InvMass;
499  Vector3 GhostLoc = this->GetLocation();
500  for ( std::list<ActorBase*>::iterator OA = OverlappingActors.begin() ; OA != OverlappingActors.end() ; OA++ )
501  {
502  if(Mezzanine::WSO_ActorRigid != (*OA)->GetType())
503  continue;
504  //Collect necessary data
505  ActorLoc = (*OA)->GetLocation();
506  Distance = ActorLoc.Distance(GhostLoc);
507  Direction = (GhostLoc - ActorLoc) / Distance;
508  switch(AttenStyle)
509  {
510  case Mezzanine::Att_Linear:
511  AppliedStrength = Strength - (AttenAmount * Distance);
512  break;
513  case Mezzanine::Att_Quadratic:
514  AppliedStrength = Strength - (AttenAmount * (Distance * Distance));
515  break;
516  default:
517  AppliedStrength = Strength;
518  break;
519  }
520  ActRig = static_cast<ActorRigid*>(*OA);
521  InvMass = ActRig->GetBulletObject()->getInvMass();
522  if(0 != InvMass)
523  AppliedStrength *= (1 / ActRig->GetBulletObject()->getInvMass());
524  else
525  AppliedStrength = 0;
526  if(0 > AppliedStrength)
527  AppliedStrength = 0;
528  //Apply the Force
529  ActRig->GetBulletObject()->applyCentralForce((Direction * AppliedStrength).GetBulletVector3());
530  }
531  }
532  if(!AllowWorldGrav && !RemovedActors.empty())
533  {
534  Vector3 WorldGrav = Entresol::GetSingletonPtr()->GetPhysicsManager()->GetGravity();
535  for ( std::vector<ActorBase*>::iterator RA = RemovedActors.begin() ; RA != RemovedActors.end() ; RA++ )
536  {
537  if(Mezzanine::WSO_ActorRigid != (*RA)->GetType())
538  continue;
539  ActRig = dynamic_cast<ActorRigid*>(*RA);
540  ActRig->GetPhysicsSettings()->SetIndividualGravity(WorldGrav);
541  }
542  }
543  }
