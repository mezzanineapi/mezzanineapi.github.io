180  void ActorRigid::_NotifyCollisionState(Physics::Collision* Col, const Physics::Collision::CollisionState& State)
181  {
182  WorldObject::_NotifyCollisionState(Col,State);
183  StickyData* StickyD = GetPhysicsSettings()->GetStickyData();
184  bool UseA = Col->GetObjectA() != this;
185  // We don't care if sticky behavior isn't set or if the collision has ended.
186  // If it's ended, then we've probably already done our logic.
187  if(0 == StickyD->MaxNumContacts || Physics::Collision::Col_End == State)
188  return;
189  // Constraints only work well with other rigid bodies, so for now just other actorrigids.
190  /// @todo Update this to be workable with other objects that have rigid bodies internally, and maybe soft bodies.
191  if( (UseA ? (Col->GetObjectA()->GetType() != Mezzanine::WSO_ActorRigid) : (Col->GetObjectB()->GetType() != Mezzanine::WSO_ActorRigid)) )
192  return;
193  // We need a contact point to be present for this to work, since a collision without contact points is an AABB overlap.
194  // So confirm there are contact points.
195  if(Col->GetNumContactPoints() < 1)
196  return;
197  // Do we have room for additional sticky constraints?
198  if(StickyD->StickyConstraints.size() < StickyD->MaxNumContacts)
199  {
200  // Does this sticky constraint already exist for this pair?
201  for( Whole X = 0 ; X < StickyD->StickyConstraints.size() ; ++X )
202  {
203  StickyConstraint* StickyCon = StickyD->StickyConstraints.at(X);
204  if( Col->PairsMatch(StickyCon->GetActorA(),StickyCon->GetActorB()) )
205  return;
206  }
207  // Alternatively, is this sticky constraint about to be constructed?
208  for( Whole Y = 0 ; Y < StickyD->CreationQueue.size() ; ++Y )
209  {
210  StickyConstraintConstructionInfo& CurrInfo = StickyD->CreationQueue.at(Y);
211  if( Col->PairsMatch(CurrInfo.ActA,CurrInfo.ActB) )
212  return;
213  }
214  // Does this collision have a contact that is actually penetrating?
215  // Does this collision have actual force?
216  bool NegativeDistFound = false;
217  Whole FoundIndex = 0;
218  Real BestMatch = 123000;
219  for( Whole X = 0 ; X < Col->GetNumContactPoints() ; ++X )
220  {
221  Real Dist = Col->GetDistance(X);
222  if( MathTools::Fabs(Dist) < BestMatch && Dist <= 0 &&
223  0 != Col->GetAppliedImpulse(X) )
224  {
225  FoundIndex = X;
226  BestMatch = MathTools::Fabs(Dist);
227  NegativeDistFound = true;
228  break;
229  }
230  }
231  if(!NegativeDistFound)
232  return;
233  // Ok, it's passed all the tests, the constraint is being made.
234  // Get the actor and it's sticky data.
235  StickyConstraintConstructionInfo NewInfo;
236  NewInfo.ActA = dynamic_cast<ActorRigid*>(UseA ? Col->GetObjectA() : Col->GetObjectB());
237  NewInfo.ActB = this;
238  //StickyData* ActorAStickyData = ActorA->GetPhysicsSettings()->GetStickyData();
239  // Calculate the transforms.
240  Vector3 ALoc = UseA ? Col->GetLocalALocation(FoundIndex) : Col->GetLocalBLocation(FoundIndex);
241  Vector3 BLoc = UseA ? Col->GetLocalBLocation(FoundIndex) : Col->GetLocalALocation(FoundIndex);
242  NewInfo.TransA = Transform(ALoc,NewInfo.ActA->GetOrientation());
243  NewInfo.TransB = Transform(BLoc,this->GetOrientation());
244  StickyD->CreationQueue.push_back(NewInfo);
245 
246  /*ObjectPair CollisionPair(ActorA,this);
247  Entresol::GetSingletonPtr()->GetPhysicsManager()->RemoveCollision(PhysicsManager::GetSingletonPtr()->GetCollision(&CollisionPair));
248 
249  // Create and configure the constraint.
250  StickyConstraint* NewSticky = new StickyConstraint(ActorA,this,TransA,TransB);
251  NewSticky->SetUpperLinLimit(0.0);
252  NewSticky->SetUpperAngLimit(0.0);
253  NewSticky->SetLowerLinLimit(0.0);
254  NewSticky->SetLowerAngLimit(0.0);
255  // Add the constraint to the world and other necessary structures.
256  Entresol::GetSingletonPtr()->GetPhysicsManager()->_GetPhysicsWorldPointer()->addConstraint(NewSticky->GetConstraintBase(),true);
257  StickyD->StickyConstraints.push_back(NewSticky);
258  ActorAStickyData->StickyConstraints.push_back(NewSticky);// */
259  }
