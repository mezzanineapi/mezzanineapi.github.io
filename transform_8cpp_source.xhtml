104  {
105  XML::Node TransformNode = CurrentRoot.AppendChild(SerializableName()); // The base node all the base constraint stuff will go in
106  if (!TransformNode)
107  { SerializeError("Create TransformNode", SerializableName()); }
108 
109  Mezzanine::XML::Attribute Version = TransformNode.AppendAttribute("Version"); // Version

111  { SerializeError("Create Version", SerializableName()); }
112  Version.SetValue(1);

114  this->Location.ProtoSerialize(TransformNode);
115  this->Rotation.ProtoSerialize(TransformNode);

118  void Transform::ProtoDeSerialize(const XML::Node& OneNode)
119  {
120  if ( Mezzanine::String(OneNode.Name())==Mezzanine::String(SerializableName()) )
121  {
122  if(OneNode.GetAttribute("Version").AsInt() == 1)
123  {
124  this->Location.ProtoDeSerialize(OneNode.GetChild("Vector3"));
125  this->Rotation.ProtoDeSerialize(OneNode.GetChild("Quaternion"));
126  }else{
127  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + SerializableName() + ": Not Version 1.");
128  }
129  }else{
130  MEZZ_EXCEPTION(Exception::II_IDENTITY_INVALID_EXCEPTION,"Attempting to deserialize a " + SerializableName() + ", found a " + String(OneNode.Name()));
131  }
132  }
