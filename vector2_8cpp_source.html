221  void Vector2::ProtoSerialize(XML::Node& CurrentRoot) const
222  {
223  Mezzanine::XML::Node VecNode = CurrentRoot.AppendChild(SerializableName());
224  VecNode.SetName(SerializableName());
225 
226  Mezzanine::XML::Attribute VersionAttr = VecNode.AppendAttribute("Version");
227  Mezzanine::XML::Attribute XAttr = VecNode.AppendAttribute("X");
228  Mezzanine::XML::Attribute YAttr = VecNode.AppendAttribute("Y");
229  if( VersionAttr && XAttr && YAttr )
230  {
231  if( VersionAttr.SetValue("1") && XAttr.SetValue(this->X) && YAttr.SetValue(this->Y) )
232  {
233  return;
234  }else{
235  SerializeError("Create XML Attribute Values", SerializableName(),true);
236  }
237  }else{
238  SerializeError("Create XML Attributes", SerializableName(),true);
239  }
