57  void ButtonDevice::UpdateButtonTransitions()
58  {
59  for( std::vector<Whole>::iterator It = TransitioningIndexes.begin() ; It != TransitioningIndexes.end() ; ++It )
60  {
61  Input::ButtonState State = Buttons.at( (*It) );
62  if( Input::BUTTON_PRESSING == State )
63  Buttons.at( (*It) ) = Input::BUTTON_DOWN;
64  else if( Input::BUTTON_LIFTING == State )
65  Buttons.at( (*It) ) = Input::BUTTON_UP;
66  }
67  TransitioningIndexes.clear();
68  }
