295 Mezzanine::XML::Node& operator >> (const Mezzanine::XML::Node& OneNode, Mezzanine::Light& Ev)
296 {
297  if ( Mezzanine::String(OneNode.Name())==Mezzanine::String("Light") )
298  {
299  if(OneNode.GetAttribute("Version").AsInt() == 1)
300  {
301  Ev.SetLightType(static_cast<Mezzanine::Light::LightType>(OneNode.GetAttribute("Type").AsInt()));
302  Ev.SetPowerScale(OneNode.GetAttribute("PowerScale").AsReal());
303  Ev.SetAttenuation(OneNode.GetAttribute("AttenuationRange").AsReal(), OneNode.GetAttribute("AttenuationConstant").AsReal(), OneNode.GetAttribute("AttenuationLinear").AsReal(), OneNode.GetAttribute("AttenuationQuadric").AsReal());
304  Ev.SetSpotlightInnerAngle(OneNode.GetAttribute("SpotlightInnerAngle").AsReal());
305  Ev.SetSpotlightOuterAngle(OneNode.GetAttribute("SpotlightOuterAngle").AsReal());
306  Ev.SetSpotlightFalloff(OneNode.GetAttribute("SpotlightFalloff").AsReal());
307  Mezzanine::WorldNode * AttachPtr = Mezzanine::Entresol::GetSingletonPtr()->GetSceneManager()->GetNode( OneNode.GetAttribute("AttachedTo").AsString() );
308  if (AttachPtr)
309  { AttachPtr->AttachObject(&Ev); }
310 
311  Mezzanine::ColourValue TempColour(0,0,0,0);
312  Mezzanine::Vector3 TempVec(0,0,0);
313 
314  for(Mezzanine::XML::Node Child = OneNode.GetFirstChild(); Child!=0; Child = Child.GetNextSibling())
315  {
316  Mezzanine::String Name(Child.Name());
317  switch(Name[0])
318  {
319  case 'f': //fDiffuseColour
320  if(Name==Mezzanine::String("fDiffuseColour"))
321  {
322  Child.GetFirstChild() >> TempColour;
323  Ev.SetDiffuseColour(TempColour);
324  }else{
325  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for Light: Includes unknown Element f-\"" + Name + "\".");
326  }
327  break;
328  case 'S': //SpecularColour
329  if(Name==Mezzanine::String("SpecularColour"))
330  {
331  Child.GetFirstChild() >> TempColour;
332  Ev.SetSpecularColour(TempColour);
333  }else{
334  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for Light: Includes unknown Element S-\"" + Name + "\".");
335  }
336  break;
337  case 'D': //Direction
338  if(Name==Mezzanine::String("Direction"))
339  {
340  Child.GetFirstChild() >> TempVec;
341  Ev.SetDirection(TempVec);
342  }else{
343  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for Light: Includes unknown Element D-\"" + Name + "\".");
344  }
345  break;
346  case 'L': //Location
347  if(Name==Mezzanine::String("Location"))
348  {
349  Child.GetFirstChild() >> TempVec;
350  Ev.SetLocation(TempVec);
351  }else{
352  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for Light: Includes unknown Element L-\"" + Name + "\".");
353  }
354  break;
355  default:
356  { MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for Light: Includes unknown Element default-\"" + Name + "\"."); }
357  break;
358  }
