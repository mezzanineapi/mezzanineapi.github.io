47  ActorRigid* ActorRigidDeSerializer::ProtoDeSerialize(const XML::Node& OneNode)
48  {
49  if ( Mezzanine::String(OneNode.Name())==ActorRigid::SerializableName() )
50  {
51  if(OneNode.GetAttribute("Version").AsInt() == 1)
52  {
53  String Name(OneNode.GetAttribute("Name").AsString());
54  if (Name.empty())
55  { DeSerializeError("find a Name Attribute",ActorRigid::SerializableName()); }
56 
57  String File(OneNode.GetAttribute("File").AsString());
58  if (File.empty())
59  { DeSerializeError("find a File Attribute",ActorRigid::SerializableName()); }
60 
61  String Group(OneNode.GetAttribute("Group").AsString());
62  if (Group.empty())
63  { DeSerializeError("find a Group Attribute",ActorRigid::SerializableName()); }
64 
65  ActorRigid *Results = new ActorRigid(10,Name,File,Group);
66  Results->ProtoDeSerialize(OneNode);
67 
68  if (0 != Target)
69  { Target->AddActor(Results); }
70 
71  if( Results->IsInWorld() != OneNode.GetChild(ActorBase::SerializableName()).GetAttribute("IsInWorld").AsBool() )
72  {
73  if(Results->IsInWorld())
74  { Results->RemoveFromWorld(); }
75  else
76  { Results->AddToWorld(); }
77  }
78 
79  /// @todo TODO: Why is gravity done here, can it be done elsewhere? Apparently the addind to the world removed what is deserialized here and this should how be used if it is different from the default
80 
81  Vector3 tempvec;
82  //[rename, /RigidActors[1]/ActorRigid[1]/ActorBase[1]/NonStaticWorldObject[1]/WorldObject[1]/ActorRigidPhysicsSettings[1]/IndividualGravity[1]/Vector3[1]/@X, LogilabXmldiffTmpAttrZ]
83 
84  XML::Node GravityNode = OneNode.GetChild(ActorBase::SerializableName()).GetChild("NonStaticWorldObject").GetChild("WorldObject").GetChild("ActorRigidPhysicsSettings").GetChild("IndividualGravity").GetFirstChild();
85  if(!GravityNode)
86  { DeSerializeError("find gravity node", "ActorRigidDeSerializer"); }
87  tempvec.ProtoDeSerialize(GravityNode);
88  Results->GetPhysicsSettings()->SetIndividualGravity(tempvec);
89 
90  return Results;
91 
92  }else{
93  DeSerializeError("find usable serialization version","ActorRigidDeSerializer");
94  }
95  }else{
96  DeSerializeError(String("find correct class to deserialize, found a ")+OneNode.Name(),"ActorRigidDeSerializer");
97  }
98  // unreachable, DeSerializeError Always throws, but this quashes that warning.
99  return 0;
100  }
