132 
133  Parent->GraphicsObject = IWOGS->WorldObjectEnt;

301  void WorldObjectGraphicsSettings::ProtoDeSerialize(const XML::Node& OneNode)
302  {
303  if ( Mezzanine::String(OneNode.Name())==this->WorldObjectGraphicsSettings::SerializableName() )
304  {
305  if(OneNode.GetAttribute("Version").AsInt() == 1)
306  {
307  Mezzanine::ColourValue TempColour;
308 
309  for(Mezzanine::Whole Counter = 0; this->GetNumSubmeshes()>Counter; ++Counter)
310  {
311  this->SetMaterialAmbient(TempColour, Counter);
312  this->SetMaterialSpecular(TempColour, Counter);
313  this->SetMaterialDiffuse(TempColour, Counter);
314  }
315 
316  for(Mezzanine::XML::Node Child = OneNode.GetFirstChild(); Child!=0; Child = Child.GetNextSibling())
317  {
318  Mezzanine::String Name(Child.Name());
319  switch(Name[0])
320  {
321  case 'A': //fDiffuseColour
322  if(Name==Mezzanine::String("AmbientMaterial"))
323  {
324  Child.GetFirstChild() >> TempColour;
325  this->SetMaterialAmbient(TempColour, Child.GetAttribute("Submesh").AsWhole());
326  }else{
327  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for WorldObjectGraphicsSettings: Includes unknown Element A-\"" + Name + "\".");
328  }
329  break;
330  case 'S': //fDiffuseColour
331  if(Name==Mezzanine::String("SpecularMaterial"))
332  {
333  Child.GetFirstChild() >> TempColour;
334  this->SetMaterialSpecular(TempColour, Child.GetAttribute("Submesh").AsWhole());
335  }else{
336  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for WorldObjectGraphicsSettings: Includes unknown Element S-\"" + Name + "\".");
337  }
338  break;
339  case 'D': //fDiffuseColour
340  if(Name==Mezzanine::String("DiffuseMaterial"))
341  {
342  Child.GetFirstChild() >> TempColour;
343  this->SetMaterialDiffuse(TempColour, Child.GetAttribute("Submesh").AsWhole());

351  }
