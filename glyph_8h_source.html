89  /// @brief Convenience function for getting the Kerning information for a given Glyph.
90  /// @param Left_of The glyph/character to the left of this one to get the Kerning information for.
91  /// @return Returns in pixels the special amount to advance.
92  inline const Real GetKerning(const char& Left_of) const
93  {
94  if(Kernings.size() == 0)
95  return 0;
96  for(size_t i=0;i < Kernings.size();i++)
97  {
98  if((UInt32)(Kernings[i].Character) == (UInt32)Left_of)
99  return Kernings[i].Kerning;
100  }
101  return 0;
102  }
