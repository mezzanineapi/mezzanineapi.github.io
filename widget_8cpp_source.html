141  void Widget::SubWidgetUpdate(bool Force)
142  {
143  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
144  if(HoveredSubWidget)
145  {
146  HoveredSubWidget->Update(Force);
147  if(Input::BUTTON_PRESSING == State)
148  {
149  SubWidgetFocus = HoveredSubWidget;
150  }
151  }
152  if(SubWidgetFocus && (SubWidgetFocus != HoveredSubWidget))
153  {
154  SubWidgetFocusUpdate(true);
155  }
156  else if(Input::BUTTON_DOWN == State && Force)
157  {
158  SubWidgetFocusUpdate(Force);
159  }
160  if(Input::BUTTON_LIFTING == State)
161  {
162  SubWidgetFocus = NULL;
163  }
164  }
