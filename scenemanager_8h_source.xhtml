141  /// @brief needs to be documented
142  enum SceneShadowTechnique
143  { //Shadow Docs from Ogre ShadowTechnique Documentation.
144  SST_None = 0, ///< No shadows.
145  SST_Stencil_Modulative = 1, ///< Stencil shadow technique which renders all shadow volumes as a modulation after all the non-transparent areas have been rendered.
146  SST_Stencil_Additive = 2, ///< Stencil shadow technique which renders each light as a separate additive pass to the scene.
147  SST_Texture_Modulative = 11, ///< Texture-based shadow technique which involves a monochrome render-to-texture of the shadow caster and a projection of that texture onto the shadow receivers as a modulative pass.
148  SST_Texture_Additive = 12, ///< Texture-based shadow technique which involves a render-to-texture of the shadow caster and a projection of that texture onto the shadow receivers, built up per light as additive passes.
149  SST_Texture_Additive_Integrated = 13, ///< Texture-based shadow technique which involves a render-to-texture of the shadow caster and a projection of that texture on to the shadow receivers, with the usage of those shadow textures completely controlled by the materials of the receivers.
150  SST_Texture_Modulative_Integrated = 14 ///< Texture-based shadow technique which involves a render-to-texture of the shadow caster and a projection of that texture on to the shadow receivers, with the usage of those shadow textures completely controlled by the materials of the receivers.

246  /// @brief Creates a skyplane for use in making a sky.
247  /// @details Only one skyplane can exist in a scene. Making a new one will remove the old one. Skyplanes are
248  /// flat planes that face in one direction. They are ideal for levels with surrounding mountains or anything
249  /// where the horizon is not visable.
250  /// @param SkyPlane The plane that will become the sky.
251  /// @param Material The name of the material to be applied to the skyplane. Note: This is not referring to the
252  /// filename, but the specific material script within the file.
253  /// @param Group The resource group where the material can be found.
254  /// @param Scale The scaling to be applied to the skyplane. This may need to be tweaked based on how high you
255  /// set the plane off the ground.
256  /// @param Tiling The number of times to tile the texture or textures listed in the material script across the skyplane.
257  /// @param DrawFirst Whether or not the skyplane should be the first thing rendered in the scene. Usually you will
258  /// want this to be true as it'll ensure all other objects are rendered on top of it.
259  /// @param Bow This will add curvature to the skyplane if set above zero. Note: Use small numbers. A bow of 1.5
260  /// should be noticable.
261  /// @param XSegments The number of segments, or boxes, the skyplane consists of on the planes X axis. This is usful
262  /// when giving the skyplane a bow. By default the skyplane is just one massive box.
263  /// @param YSegments The number of segments, or boxes, the skyplane consists of on the planes Y axis. This is usful
264  /// when giving the skyplane a bow. By default the skyplane is just one massive box.
265  void CreateSkyPlane(const Plane& SkyPlane_, const String& Material, const String& Group, Real Scale=1000.0, Real Tiling=10.0,
266  bool DrawFirst=true, Real Bow=0, int XSegments=1, int YSegments=1);
267  /// @brief Disables the currently active skyplane.
268  /// @details Using this function effectively deletes the skyplane, so you will have to provide a new set of parameters
269  /// if you wish to re-create the skyplane.
270  void DisableSkyPlane();
271 
272  /// @brief Creates a skybox for use in making a sky.
273  /// @details Like skyplanes, only one can exist per scene. Unlike skyplanes, skyboxes will be applied individually to
274  /// each camera in the scene. The skybox will move with the camera, so as a result the camera will never be able to
275  /// "touch" the sky. Skyboxes are more performance intensive then skyplanes.
276  /// @param Material The name of the material to be applied to the skybox. Note: This is not referring to the
277  /// filename, but the specific material script within the file.
278  /// @param Group The resource group where the material can be found.
279  /// @param Distance The distance from the camera where the skybox is found. This is in world units.
280  /// @param DrawFirst Whether or not the skybox should be the first thing rendered in the scene. Usually you will
281  /// want this to be true as it'll ensure all other objects are rendered on top of it.
282  /// @param Orientation Optional quaternion to rotate the orientation of the skybox.
283  void CreateSkyBox(const String& Material, const String& Group, Real Distance, bool DrawFirst=true, Quaternion Orientation=Quaternion());
284  /// @brief Disables the currently active skybox.
