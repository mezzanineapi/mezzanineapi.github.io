166  void TextLayer::RedrawImpl(bool Force)
167  {
168  // Update our text
169  if(Force || this->ScalingChanged)
170  {
171  // In general the only time this will be forced is on a resizing of the parent
172  // quad renderable. So we must repopulate for the change of space.
173  this->PopulateTextLines();
174  this->ScalingChanged = false;
175  }
176 
177  // Get the Texel Offsets
178  Real TexelOffsetX = this->Parent->GetScreen()->GetTexelOffsetX();
179  Real TexelOffsetY = this->Parent->GetScreen()->GetTexelOffsetY();
180 
181  // Get the parent rect and apply the scaling
182  Rect ActDims = this->GetAreaRect();
183 
184  // Setup the text specific data we'll use
185  Vector2 TopLeft, TopRight, BottomLeft, BottomRight;
186  Vector2 WhitePixelCoords;
187  Real CursorX = ActDims.Position.X, CursorY = ActDims.Position.Y;
188  ColourValue HighlightColour = this->InactiveHLColour;
189  Character* CurrChar = NULL;
190  TextLine* CurrLine = NULL;
191 
192  // Check if we need the active highlight colour
193  if( Renderable::RT_Widget == this->Parent->GetRenderableType() && (static_cast<Widget*>(this->Parent))->HasFocus() )
194  HighlightColour = ActiveHLColour;
195 
196  for( TextLineIterator LineIt = this->TextLines.begin() ; LineIt != this->TextLines.end() ; ++LineIt )
197  {
198  CurrLine = (*LineIt);
199  CursorY = ActDims.Position.Y + CurrLine->GetPositionOffset();
200 
201  // If we have ran out of room, halt rendering
202  if( CursorY >= ActDims.Position.Y + ActDims.Size.Y )
203  break;
204 
205  // Update the X cursor starting position
206  CursorX = ActDims.Position.X + CurrLine->GetCursorStartPosition();
207 
208  for( TextLine::CharacterIterator CharIt = CurrLine->BeginCharacter() ; CharIt != CurrLine->EndCharacter() ; ++CharIt )
209  {
210  CurrChar = (*CharIt);
211  if( !CurrChar->IsRenderable() )
212  continue;
213 
214  String CharAtlas = CurrChar->GetAtlasName();
215  Real VertOffset = CurrChar->GetVerticalOffset() * this->CharScaling.Y;
216  Vector2 CharSize = CurrChar->GetCharacterSize() * this->CharScaling;
217  ColourValue CharColour = CurrChar->GetCharacterColour();
218  CursorX = ActDims.Position.X + CurrChar->GetLengthOffset();
219 
220  TopLeft.SetValues( MathTools::Floor( CursorX - TexelOffsetX ),
221  MathTools::Floor( (CursorY + TexelOffsetY + VertOffset) - CharSize.Y ) );
222  BottomRight.SetValues( MathTools::Floor( (CursorX - TexelOffsetX) + CharSize.X ),
223  MathTools::Floor( CursorY + TexelOffsetY + VertOffset ) );
224  TopRight.SetValues( BottomRight.X, TopLeft.Y );
225  BottomLeft.SetValues( TopLeft.X, BottomRight.Y );
226 
227  this->RotationTransform(TopLeft,TopRight,BottomLeft,BottomRight);
228 
229  if( CurrChar->GetHighlighted() )
230  {
231  WhitePixelCoords = CurrChar->GetAtlasWhitePixel();
232 
233  // Triangle A
234  PushVertex(BottomLeft.X, BottomLeft.Y, WhitePixelCoords, HighlightColour, CharAtlas); // Left/Bottom 3
235  PushVertex(TopRight.X, TopRight.Y, WhitePixelCoords, HighlightColour, CharAtlas); // Right/Top 1
236  PushVertex(TopLeft.X, TopLeft.Y, WhitePixelCoords, HighlightColour, CharAtlas); // Left/Top 0
237 
238  // Triangle B
239  PushVertex(BottomLeft.X, BottomLeft.Y, WhitePixelCoords, HighlightColour, CharAtlas); // Left/Bottom 3
240  PushVertex(BottomRight.X, BottomRight.Y, WhitePixelCoords, HighlightColour, CharAtlas); // Right/Bottom 2
241  PushVertex(TopRight.X, TopRight.Y, WhitePixelCoords, HighlightColour, CharAtlas); // Right/Top 1
242  }
243 
244  // Triangle A
245  PushVertex(BottomLeft.X, BottomLeft.Y, CurrChar->GetRelativeAtlasCoords(UI::QC_BottomLeft), CharColour, CharAtlas); // Left/Bottom 3
246  PushVertex(TopRight.X, TopRight.Y, CurrChar->GetRelativeAtlasCoords(UI::QC_TopRight), CharColour, CharAtlas); // Right/Top 1
247  PushVertex(TopLeft.X, TopLeft.Y, CurrChar->GetRelativeAtlasCoords(UI::QC_TopLeft), CharColour, CharAtlas); // Left/Top 0
248 
249  // Triangle B
250  PushVertex(BottomLeft.X, BottomLeft.Y, CurrChar->GetRelativeAtlasCoords(UI::QC_BottomLeft), CharColour, CharAtlas); // Left/Bottom 3
251  PushVertex(BottomRight.X, BottomRight.Y, CurrChar->GetRelativeAtlasCoords(UI::QC_BottomRight), CharColour, CharAtlas); // Right/Bottom 2
252  PushVertex(TopRight.X, TopRight.Y, CurrChar->GetRelativeAtlasCoords(UI::QC_TopRight), CharColour, CharAtlas); // Right/Top 1
253  }//for each character
254  }//for each text line
255 
256  // Last thing to do is rendering the text cursor
257  if( this->GetCursorEnabled() ) {
258  if( this->Cursor->GetVisible() ) {
259  Rect CursorRect = Cursor->GetCursorRect();
260  ColourValue CursorColour = Cursor->GetColour();
261  WhitePixelCoords = Parent->GetScreen()->GetWhitePixel(PriAtlas);
262 
263  BottomLeft = ActDims.Position + CursorRect.Position;
264  TopRight.SetValues( BottomLeft.X + CursorRect.Size.X , BottomLeft.Y - CursorRect.Size.Y );
265  TopLeft.SetValues( BottomLeft.X , TopRight.Y );
266  BottomRight.SetValues( TopRight.X , BottomLeft.Y );
267 
268  // Triangle A
269  PushVertex(BottomLeft.X, BottomLeft.Y, WhitePixelCoords, CursorColour, PriAtlas); // Left/Bottom 3
270  PushVertex(TopRight.X, TopRight.Y, WhitePixelCoords, CursorColour, PriAtlas); // Right/Top 1
271  PushVertex(TopLeft.X, TopLeft.Y, WhitePixelCoords, CursorColour, PriAtlas); // Left/Top 0
272 
273  // Triangle B
274  PushVertex(BottomLeft.X, BottomLeft.Y, WhitePixelCoords, CursorColour, PriAtlas); // Left/Bottom 3
275  PushVertex(BottomRight.X, BottomRight.Y, WhitePixelCoords, CursorColour, PriAtlas); // Right/Bottom 2
276  PushVertex(TopRight.X, TopRight.Y, WhitePixelCoords, CursorColour, PriAtlas); // Right/Top 1
277  }
278  }
279  }

291  void TextLayer::RecalculateOffsets()
292  {
293  // Get the actual starting position on the Y axis
294  Real CursorY = 0;
295  Real TotalHeight = this->GetTotalHeight();
296  Rect ActDims = this->GetAreaRect();
297  if( TotalHeight < ActDims.Size.Y )
298  {
299  switch(VerticalAlign)
300  {
301  case UI::LA_TopLeft: /* do nothing, we're at the top */ break;
302  case UI::LA_BottomRight: CursorY = ActDims.Size.Y - TotalHeight; break;
303  case UI::LA_Center: CursorY = (ActDims.Size.Y * 0.5) - (TotalHeight * 0.5); break;
304  }
305  }
306  // Set all the actual offsets
307  for( TextLineIterator TLIt = TextLines.begin() ; TLIt != TextLines.end() ; ++TLIt )
308  {
309  CursorY += (*TLIt)->GetLineHeight();
310  (*TLIt)->SetPositionOffset(CursorY);
311  }
312  }

369  void TextLayer::SetText(const String& Text)
370  {
371  if( this->DefaultCharTraits.CharFont == NULL ) {
372  StringStream ExceptionStream;
373  ExceptionStream << "Attempting to set text on a TextLayer without a default font set. Layer owned by renderable \"" << Parent->GetName() << "\".";
374  MEZZ_EXCEPTION(Exception::INVALID_STATE_EXCEPTION,ExceptionStream.str());
375  }
376 
377  /// @todo Character generation and storage is a good candidate for pool allocation.
378  this->DestroyAllCharacters();
379 
380  if( this->TextTokens != NULL ) {
381  delete this->TextTokens;
382  this->TextTokens = NULL;
383  }
384 
385  this->TextTokens = this->MUParser->Lex(Text);
386  this->Characters = this->MUParser->ParseTextTokens(this->TextTokens,this->DefaultCharTraits,this);
387  this->PopulateTextLines();
388 
389  this->_MarkDirty();
390  }

621  TextLine* TextLayer::GetTextLineAtOffset(const Real& Offset)
622  {
623  TextLineIterator LineIt = TextLines.begin();
624  // Check if we're too far to the top to get anything
625  if( Offset < (*LineIt)->GetPositionOffset() - (*LineIt)->GetLineHeight() )
626  return NULL;
627 
628  while( LineIt != this->TextLines.end() && Offset < (*LineIt)->GetPositionOffset() )
629  {
630  ++LineIt;
631  }
632 
633  // Check if we're too low to get anything
634  if( LineIt == (--this->TextLines.end()) && Offset > (*LineIt)->GetPositionOffset() )
635  return NULL;
636 
637  return (*LineIt);
638  }

715  Character* TextLayer::GetCharacterAtOffset(const Vector2& Offset) const
716  {
717  // Get the parent rect and apply the scaling
718  Rect ActDims = this->GetAreaRect();
719  if( !ActDims.IsInside(ActDims.Position + Offset) || this->TextLines.empty() )
720  return NULL;
721 
722  // Get our line
723  ConstTextLineIterator LineIt = this->TextLines.begin();
724  while( LineIt != this->TextLines.end() && Offset.Y > (*LineIt)->GetPositionOffset() )
725  {
726  /// @todo The way this logic works, in the event of a slight bit of clipping at the bottom of the quad space that
727  /// isn't enough for the next line, this logic will assume that space is a part of the last line. Is that a problem?
728  ++LineIt;
729  }
730 
731  // Quickly check to see if we're to the side of any lines.
732  if( Offset.X < (*LineIt)->GetLeftMostCursorPosition() ||
733  Offset.X > (*LineIt)->GetRightMostCursorPosition() )
734  {
735  return NULL;
736  }
737 
738  // Get the character in that line
739  CharacterIterator CharIt = (*LineIt)->BeginCharacter();
740  while( CharIt != (*LineIt)->EndCharacter() && Offset.X > (*CharIt)->GetLengthOffset() + (*CharIt)->GetCharacterSize().X )
741  {
742  /// @todo In the event that letter spacing for the font is big ( >0 ), and the point provided rests on a gap in the spacing,
743  /// this algorithm will grab the character to the right as the hovered character. Is that a problem?
744  ++CharIt;
745  }
746 
747  if( CharIt == (*LineIt)->EndCharacter() ) return NULL;
748  else return (*CharIt);
749  }
