61  : Widget(name,parent),

111  CreateVerticalSpinner(APos,ASize,GlyphHeight);

115  }

152  void Spinner::CreateHorizontalSpinner(const Vector2& Position, const Vector2& Size, const Real& GlyphHeight)
153  {
154  const Vector2& WinDim = ParentScreen->GetViewportDimensions();
155  Whole ActHeight = (Whole)(GlyphHeight * WinDim.Y);
156  std::pair<Whole,Real> GlyphInfo = Manager->SuggestGlyphIndex(ActHeight,ParentScreen->GetPrimaryAtlas());
157  Vector2 IncPos = Position + IncrementOffset;
158  Vector2 DecPos = Position + DecrementOffset;
159  Vector2 ValPos = Position + ValueDisplayOffset;
160  Increment = ParentScreen->CreateButton(Name+"Inc",RenderableRect(IncPos,Vector2(Size.Y,Size.Y),false));
161  Decrement = ParentScreen->CreateButton(Name+"Dec",RenderableRect(DecPos,Vector2(Size.Y,Size.Y),false));
162  ValueDisplay = ParentScreen->CreateCaption(Name+"Dis",RenderableRect(ValPos,Vector2(Size.X - (Size.Y * 2),Size.Y),false),GlyphInfo.first,GetValueAsText());
163  AddSubRenderable(0,Increment);
164  AddSubRenderable(1,Decrement);
165  AddSubRenderable(2,ValueDisplay);
166  if(1 != GlyphInfo.second)
167  ValueDisplay->SetTextScale(GlyphInfo.second);
168  }

206  void Spinner::CalculateOffsets(const Vector2& Size)
207  {
208  switch (SpinLayout)
209  {
210  case UI::Spn_Separate:
211  {
212  if(Size.X > Size.Y * 2)
213  {
214  IncrementOffset = Vector2(Size.X - Size.Y,0);
215  DecrementOffset = Vector2(0,0);
216  ValueDisplayOffset = Vector2(Size.Y,0);
217  }
218  else if(Size.Y > Size.X * 2)
219  {
220  IncrementOffset = Vector2(0,0);
221  DecrementOffset = Vector2(0,Size.Y - Size.X);
222  ValueDisplayOffset = Vector2(0,Size.X);
223  }
224  break;
225  }
226  case UI::Spn_Together_Left:
227  {
228  IncrementOffset = Vector2(0,0);
229  DecrementOffset = Vector2(0,Size.Y * 0.5);
230  ValueDisplayOffset = Vector2(Size.Y * 0.5,0);
231  break;
232  }
233  case UI::Spn_Together_Right:
234  {
235  IncrementOffset = Vector2(Size.X - (Size.Y * 0.5),0);
236  DecrementOffset = Vector2(Size.X - (Size.Y * 0.5),Size.Y * 0.5);
237  ValueDisplayOffset = Vector2(0,0);
238  break;
239  }
240  }
241  }

305  void Spinner::UpdateImpl(bool Force)
306  {
307  ProcessCapturedInputs();
308  if( HoveredSubWidget && (Widget::W_Button == HoveredSubWidget->GetType()) )
309  {
310  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
311  if(Input::BUTTON_PRESSING == State)
312  {
313  if(HoveredSubWidget==Increment)
314  IncrementLock = false;
315  else if(HoveredSubWidget==Decrement)
316  DecrementLock = false;
317  }
318  else if(Input::BUTTON_LIFTING && (!IncrementLock || !DecrementLock))
319  {
320  if(HoveredSubWidget==Increment && !IncrementLock)
321  {
322  Value++;
323  CheckValueLimits();
324  ValueDisplay->SetText(GetValueAsText());
325  IncrementLock = true;
326  }
327  else if(HoveredSubWidget==Decrement && !DecrementLock)
328  {
329  Value--;
330  CheckValueLimits();
331  ValueDisplay->SetText(GetValueAsText());
332  DecrementLock = true;
333  }
334  }
335  }
336  }

365  void Spinner::ProcessCapturedInputs()
366  {
367  if(!Editable)
368  return;
369  if(0 == CaptureData->GetNumCapturedInputs())
370  return;
371  std::vector<Input::InputCode>* CCodes = CaptureData->GetCapturedInputs();
372  bool EnterPressed = false;
373  for( Whole X = 0 ; X < CCodes->size() ; X++ )
374  {
375  Input::InputCode Current = CCodes->at(X);
376  if(Input::KEY_RETURN == Current || Input::KEY_KP_ENTER == Current)
377  {
378  EnterPressed = true;
379  continue;
380  }else{
381  EditCache+=StringTools::ConvertToString(Current,false);
382  }
383  }
384  if(EnterPressed)
385  {
386  std::stringstream TempStream;
387  if(DecimalDisplay)
388  EditCache.erase(EditCache.find("."),1);
389  TempStream << EditCache;
390  TempStream >> Value;
391  CheckValueLimits();
392  ValueDisplay->SetText(GetValueAsText());
393  EditCache.clear();
394  }
395  }
