98  void RenderLayerGroup::AddLayer(RenderLayer* RL, const UInt16 ZOrder)
99  {
100  for( RenderLayerIterator It = this->RenderLayers.begin() ; It != this->RenderLayers.end() ; ++It )
101  {
102  if( (*It).first > ZOrder ) {
103  this->RenderLayers.insert(It,RenderLayerPair(ZOrder,RL));
104  if( this->ParentQuad->GetActiveGroup() == this ) {
105  this->ParentQuad->_MarkDirty();
106  }
107  return;
108  }
109  }
110  this->RenderLayers.push_back(RenderLayerPair(ZOrder,RL));
111  if( this->ParentQuad->GetActiveGroup() == this ) {
112  this->ParentQuad->_MarkDirty();
113  }
114  return;
115  }

604  void QuadRenderable::DestroyRenderLayer(RenderLayer* ToBeDestroyed)
605  {
606  for( RenderLayerGroupIterator It = this->RenderLayerGroups.begin() ; It != this->RenderLayerGroups.end() ; ++It )
607  {
608  (*It).second->RemoveLayer(ToBeDestroyed);
609  }
610  for( Whole Index = 0 ; Index < this->RenderLayers.size() ; ++Index )
611  {
612  if( ToBeDestroyed == this->RenderLayers[Index] ) {
613  if( Index != this->RenderLayers.size() - 1 ) {
614  // swap...
615  std::swap(this->RenderLayers[Index],this->RenderLayers[this->RenderLayers.size()-1]);
616  this->RenderLayers[Index]->_UpdateIndex(Index);
617  }
618  // ...and pop
619  this->RenderLayers.pop_back();
620  break;
621  }
622  }
623  delete ToBeDestroyed;
624  }
625 
626  void QuadRenderable::DestroyAllRenderLayers()
627  {
628  for( RenderLayerGroupIterator It = this->RenderLayerGroups.begin() ; It != this->RenderLayerGroups.end() ; ++It )
629  {

631  }
632  for( RenderLayerIterator It = this->RenderLayers.begin() ; It != this->RenderLayers.end() ; ++It )

635  }
636  this->RenderLayers.clear();
637  }
