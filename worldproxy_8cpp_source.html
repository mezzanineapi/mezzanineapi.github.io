79  void WorldProxy::ProtoSerializeProperties(XML::Node& SelfRoot) const
80  {
81  XML::Node PropertiesNode = SelfRoot.AppendChild( WorldProxy::GetSerializableName() + "Properties" );
82 
83  if( PropertiesNode.AppendAttribute("Version").SetValue("1") )

87  XML::Node OrientationNode = PropertiesNode.AppendChild("Orientation");
88  this->GetOrientation().ProtoSerialize( OrientationNode );

98  void WorldProxy::ProtoDeSerialize(const XML::Node& SelfRoot)
99  {
100  Boolean WasInWorld = false;
101  XML::Attribute InWorldAttrib = SelfRoot.GetAttribute("InWorld");
102  if( !InWorldAttrib.Empty() ) {
103  WasInWorld = StringTools::ConvertToBool( InWorldAttrib.AsString() );
104  }
105 
106  this->ProtoDeSerializeProperties(SelfRoot);
107 
108  if( WasInWorld ) {
109  this->AddToWorld();
110  }

113  void WorldProxy::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
114  {
115  XML::Attribute CurrAttrib;
116  XML::Node PropertiesNode = SelfRoot.GetChild( WorldProxy::GetSerializableName() + "Properties" );
117 
118  if( !PropertiesNode.Empty() ) {
119  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
120  XML::Node PositionNode = PropertiesNode.GetChild("Location").GetFirstChild();
121  if( !PositionNode.Empty() ) {
122  Vector3 Loc(PositionNode);
123  this->SetLocation(Loc);
124  }
125 
126  XML::Node OrientationNode = PropertiesNode.GetChild("Orientation").GetFirstChild();
127  if( !PositionNode.Empty() ) {
128  Quaternion Rot(OrientationNode);
129  this->SetOrientation(Rot);
130  }
131 
132  XML::Node ScaleNode = PropertiesNode.GetChild("Scale").GetFirstChild();
133  if( !PositionNode.Empty() ) {
134  Vector3 Scale(ScaleNode);
135  this->SetScale(Scale);
136  }
137  }else{
138  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (WorldProxy::GetSerializableName() + "Properties" ) + ": Not Version 1.");
139  }
140  }else{
141  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,WorldProxy::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
142  }
143  }
