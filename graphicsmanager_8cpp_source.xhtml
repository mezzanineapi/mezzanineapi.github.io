124  GraphicsManager::GraphicsManager(XML::Node& XMLNode) :
125  RenderWork(NULL),
126  ThreadResources(NULL),
127  OgreBeenInitialized(false),
128  CurrRenderSys(Graphics::RS_OpenGL2)
129  {
130  this->Construct();
131 
132  XML::Attribute CurrAttrib;
133  String PathPreset;
134  // Get whether or not to autogen the directory path and settings file.
135  XML::Node AutoGenNode = XMLNode.GetChild("AutoCreateSettings");
136  if(!AutoGenNode.Empty())
137  {
138  CurrAttrib = AutoGenNode.GetAttribute("Auto");
139  if(!CurrAttrib.Empty())
140  this->AutoGenPath = this->AutoGenFiles = StringTools::ConvertToBool( CurrAttrib.AsString() );
141  }
142  // Get preset path to default to if a path is not provided.
143  XML::Node PathNode = XMLNode.GetChild("SettingsPath");
144  if(!PathNode.Empty())
145  {
146  CurrAttrib = PathNode.GetAttribute("Path");
147  if(!CurrAttrib.Empty())
148  PathPreset = CurrAttrib.AsString();
149 
150  if(!PathPreset.empty())
151  this->SetSettingsFilePath(PathPreset);
152  }
153  // Get the files to be loaded, and load them.
154  XML::Node FilesNode = XMLNode.GetChild("SettingsFiles");
155  if(!FilesNode.Empty())
156  {
157  for( XML::NodeIterator SetFileIt = FilesNode.begin() ; SetFileIt != FilesNode.end() ; ++SetFileIt )
158  {
159  String FileName, FilePath, FileGroup;
160  // Get the filename to load
161  CurrAttrib = (*SetFileIt).GetAttribute("FileName");
162  if(!CurrAttrib.Empty())
163  FileName = CurrAttrib.AsString();
164  // Get the path
165  CurrAttrib = (*SetFileIt).GetAttribute("Path");
166  if(!CurrAttrib.Empty())
167  FilePath = CurrAttrib.AsString();
168  else
169  {
170  CurrAttrib = (*SetFileIt).GetAttribute("Group");
171  if(!CurrAttrib.Empty())
172  FileGroup = CurrAttrib.AsString();
173  }
174 
175  if(FilePath.empty())
176  {
177  if(FileGroup.empty()) this->LoadSettings(FileName);
178  else this->LoadSettingsFromGroup(FileName,FileGroup);
179  }
180  else this->LoadSettings(FileName,FilePath);
181  }
182  }
183  /// @todo This is currently necessary because a render window of some kind needs to exist for the loading
184  /// of resources that occurs later in world construction (when this is constructed by the world, which this
185  /// assumes. If possible this should be removed, to keep construction more flexible.
186  this->InitOgreRenderSystem();
187  }

269  }
270 
271  OgreCore->initialise(false,"");
272  this->OgreBeenInitialized = true;
273 
274  this->PrimaryGameWindow = new Graphics::GameWindow("Primary",1,1,Graphics::GameWindow::WF_Hidden);

301  WindowConfigNode.AppendAttribute("Resizeable").SetValue( StringTools::ConvertToString( (*WinIt)->BorderIsResizeable() ) );
302  WindowConfigNode.AppendAttribute("Borderless").SetValue( StringTools::ConvertToString( (*WinIt)->IsBorderless() ) );
303  WindowConfigNode.AppendAttribute("FSAA").SetValue( StringTools::ConvertToString( (*WinIt)->GetFSAALevel() ) );
304  /// @todo Currently the maximized setting does nothing in the gamewindow. If it gets implemented, so does this.
305  //WindowConfigNode.AppendAttribute("Maximized").SetValue( (*WinIt)-> );//

307  {
308  XML::Node ViewportConfigNode = WindowConfigNode.AppendChild("Viewport");
309  ViewportConfigNode.AppendAttribute("ZOrder").SetValue( (*VPIt).second->GetZOrder() );

311  ViewportConfigNode.AppendAttribute("Size").SetValue( StringTools::ConvertToString( Vector2((*VPIt).second->GetWidth(),(*VPIt).second->GetHeight()) ) );
312  }

316  void GraphicsManager::ApplySettingGroupImpl(ObjectSettingGroup* Group)
317  {
318  for( ObjectSettingSetContainer::SubSetIterator SubSetIt = Group->SubSetBegin() ; SubSetIt != Group->SubSetEnd() ; ++SubSetIt )
319  {
320  String CurrSettingValue;
321  if( "RenderSystem" == (*SubSetIt)->GetName() )
322  {
323  Graphics::RenderSystem RenderSys = Graphics::RS_OpenGL2;
324  CurrSettingValue = (*SubSetIt)->GetSettingValue("Name");
325  if( GetShortenedRenderSystemName(Graphics::RS_DirectX9) == CurrSettingValue )
326  RenderSys = Graphics::RS_DirectX9;
327  else if( GetShortenedRenderSystemName(Graphics::RS_DirectX11) == CurrSettingValue )
328  RenderSys = Graphics::RS_DirectX11;
329  else if( GetShortenedRenderSystemName(Graphics::RS_OpenGL2) == CurrSettingValue )
330  RenderSys = Graphics::RS_OpenGL2;
331  else if( GetShortenedRenderSystemName(Graphics::RS_OpenGLES1) == CurrSettingValue )
332  RenderSys = Graphics::RS_OpenGLES1;
333  else if( GetShortenedRenderSystemName(Graphics::RS_OpenGLES2) == CurrSettingValue )
334  RenderSys = Graphics::RS_OpenGLES2;
335 
336  this->CurrRenderSys = RenderSys;
337 
338  if( !this->OgreBeenInitialized )
339  {
340  this->SetRenderSystem(this->CurrRenderSys,true);
341  }else{
342  /// @todo May want to make some other data member so that people can accurately get what is set now, instead of what will be set.
343  Entresol::GetSingletonPtr()->Log("WARNING: Attempting to apply new RenderSystem settings after the GraphicsManager has been initialized. "
344  "These Settings will be applied the next time settings are loaded during manager construction if current settings are saved.");
345  }
346  }
347  else if( "GameWindow" == (*SubSetIt)->GetName() )
348  {
349  String WinCaption("Mezzanine Window");
350  Whole WinWidth = 800;
351  Whole WinHeight = 600;
352  Whole WinFlags = 0;
353 
354  // Get the caption.
355  CurrSettingValue = (*SubSetIt)->GetSettingValue("Caption");
356  if(!CurrSettingValue.empty())
357  WinCaption = CurrSettingValue;
358  // Get the width.
359  CurrSettingValue = (*SubSetIt)->GetSettingValue("Width");
360  if(!CurrSettingValue.empty())
361  WinWidth = StringTools::ConvertToUInt32(CurrSettingValue);
362  // Get the height.
363  CurrSettingValue = (*SubSetIt)->GetSettingValue("Height");
364  if(!CurrSettingValue.empty())
365  WinHeight = StringTools::ConvertToUInt32(CurrSettingValue);
366  // Get fullscreen.
367  CurrSettingValue = (*SubSetIt)->GetSettingValue("Fullscreen");
368  if(!CurrSettingValue.empty())
369  {
370  if(StringTools::ConvertToBool(CurrSettingValue))
371  WinFlags = (WinFlags | Graphics::GameWindow::WF_Fullscreen);
372  }
373  // Get hidden.
374  CurrSettingValue = (*SubSetIt)->GetSettingValue("Hidden");
375  if(!CurrSettingValue.empty())
376  {
377  if(StringTools::ConvertToBool(CurrSettingValue))
378  WinFlags = (WinFlags | Graphics::GameWindow::WF_Hidden);
379  }
380  // Get vsync.
381  CurrSettingValue = (*SubSetIt)->GetSettingValue("Vsync");
382  if(!CurrSettingValue.empty())
383  {
384  if(StringTools::ConvertToBool(CurrSettingValue))
385  WinFlags = (WinFlags | Graphics::GameWindow::WF_VsyncEnabled);
386  }
387  // Get resizable.
388  CurrSettingValue = (*SubSetIt)->GetSettingValue("Resizeable");
389  if(!CurrSettingValue.empty())
390  {
391  if(StringTools::ConvertToBool(CurrSettingValue))
392  WinFlags = (WinFlags | Graphics::GameWindow::WF_Resizeable);
393  }
394  // Get maximized.
395  CurrSettingValue = (*SubSetIt)->GetSettingValue("Maximized");
396  if(!CurrSettingValue.empty())
397  {
398  if(StringTools::ConvertToBool(CurrSettingValue))
399  WinFlags = (WinFlags | Graphics::GameWindow::WF_Maximized);
400  }
401  // Get borderless.
402  CurrSettingValue = (*SubSetIt)->GetSettingValue("Borderless");
403  if(!CurrSettingValue.empty())
404  {
405  if(StringTools::ConvertToBool(CurrSettingValue))
406  WinFlags = (WinFlags | Graphics::GameWindow::WF_Borderless);
407  }
408  // Get the FSAA level
409  CurrSettingValue = (*SubSetIt)->GetSettingValue("FSAA");
410  if(!CurrSettingValue.empty())
411  {
412  switch (StringTools::ConvertToUInt32(CurrSettingValue))
413  {
414  case 2:
415  WinFlags = (WinFlags | Graphics::GameWindow::WF_FSAA_2);
416  break;
417  case 4:
418  WinFlags = (WinFlags | Graphics::GameWindow::WF_FSAA_4);
419  break;
420  case 8:
421  WinFlags = (WinFlags | Graphics::GameWindow::WF_FSAA_8);
422  break;
423  case 16:
424  WinFlags = (WinFlags | Graphics::GameWindow::WF_FSAA_16);
425  break;
426  }
427  }
428  // Finally, construct the window.
429  Graphics::GameWindow* CurrWindow = this->CreateGameWindow(WinCaption,WinWidth,WinHeight,WinFlags);
430  // Set up the viewports
431  for( ObjectSettingSetContainer::SubSetIterator VPIt = (*SubSetIt)->SubSetBegin() ; VPIt != (*SubSetIt)->SubSetEnd() ; ++VPIt )
432  {
433  if( "Viewport" == (*VPIt)->GetName() )
434  {
435  Integer ZO = 0;
436  Vector2 Position(1,1);
437  Vector2 Size(0,0);
438 
439  CurrSettingValue = (*VPIt)->GetSettingValue("ZOrder");
440  if(!CurrSettingValue.empty())
441  ZO = StringTools::ConvertToInteger( CurrSettingValue );
442  CurrSettingValue = (*VPIt)->GetSettingValue("Position");
443  if(!CurrSettingValue.empty())
444  Position = StringTools::ConvertToVector2( CurrSettingValue );
445  CurrSettingValue = (*VPIt)->GetSettingValue("Size");
446  if(!CurrSettingValue.empty())
447  Size = StringTools::ConvertToVector2( CurrSettingValue );
448 
449  Graphics::Viewport* CurrViewport = CurrWindow->CreateViewport(NULL,ZO);
450  CurrViewport->SetDimensions(Position.X,Position.Y,Size.X,Size.Y);
451  }// if - Viewport
452  }// for - Viewports
453  }// if - RS || GameWindow
454  }// for - SubSets
455  }
