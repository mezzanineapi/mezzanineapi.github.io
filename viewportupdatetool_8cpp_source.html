127  Vector2 ViewportUpdateTool::RPM_ScaleQuadrentsByHeight(const Vector2& RendPosition, const Vector2& RendSize, const Vector2& UpdatedRendSize)
128  {
129  Vector2 NewRendPos;
130  Vector2 RelPos = RendPosition / OldSize;
131  Vector2 Middle;
132  Middle.X = RendPosition.X + (RendSize.X * 0.5);
133  Middle.Y = RendPosition.Y + (RendSize.Y * 0.5);
134 
135  bool Left = (Middle.X < (OldSize.X - (MiddleTolerance * OldSize.X)) * 0.5);

137  bool Top = (Middle.Y < (OldSize.Y - (MiddleTolerance * OldSize.Y)) * 0.5);
138  bool Bottom = (Middle.Y > (OldSize.Y + (MiddleTolerance * OldSize.Y)) * 0.5);
139 
140  Real Scale = NewSize.Y / OldSize.Y;
141 
142  if(Top)
143  {
144  Real RelDist = RelPos.Y;
145  NewRendPos.Y = RelDist * NewSize.Y;

147  Real RelDist = (OldSize.Y - (RendPosition.Y + RendSize.Y)) / OldSize.Y;
148  NewRendPos.Y = (NewSize.Y - (NewSize.Y * RelDist)) - UpdatedRendSize.Y;
149  }else{
150  Real RelMiddle = Middle.Y / OldSize.Y;
151  NewRendPos.Y = (RelMiddle * NewSize.Y) - (UpdatedRendSize.Y * 0.5);
152  }
153 
154  if(Left)
155  {
156  NewRendPos.X = RendPosition.X * Scale;
157  }else if(Right){
158  Real Dist = OldSize.X - (RendPosition.X + RendSize.X);
159  NewRendPos.X = (NewSize.X - (Dist * Scale)) - UpdatedRendSize.X;
160  }else{
161  Real ScreenMiddle = OldSize.X * 0.5;
162  if(Middle.X < ScreenMiddle - 1) //left
163  {
164  Real MidDiff = ((RendPosition.X + RendSize.X) - ScreenMiddle) * Scale;
165  NewRendPos.X = ((NewSize.X * 0.5) + MidDiff) - UpdatedRendSize.X;
166  }else if(Middle.X > ScreenMiddle + 1){ //right
167  Real MidDiff = (RendPosition.X - ScreenMiddle) * Scale;
168  NewRendPos.X = (NewSize.X * 0.5) + MidDiff;
169  }else{
170  Real RelMiddle = Middle.X / OldSize.X;
171  NewRendPos.X = (RelMiddle * NewSize.X) - (UpdatedRendSize.X * 0.5);
172  }
173  }
