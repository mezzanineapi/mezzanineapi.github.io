238  void AudioManager::DestroyDecoderFactory(const Audio::Encoding Encode)
239  {
240  DecoderFactoryIterator FactIt = this->DecoderFactories.find(Encode);
241  if( FactIt != this->DecoderFactories.end() )
242  {
243  delete (*FactIt).second;
244  this->DecoderFactories.erase(FactIt);
245  }
246  }

248  void AudioManager::DestroyAllDecoderFactories()
249  {
250  for( DecoderFactoryIterator FactIt = this->DecoderFactories.begin() ; FactIt != this->DecoderFactories.end() ; ++FactIt )
251  {
252  delete (*FactIt).second;
253  }
254  this->DecoderFactories.clear();
255  }
