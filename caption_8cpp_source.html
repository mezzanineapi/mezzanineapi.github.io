95  void Caption::CalculateDrawSize(Vector2& Size)
96  {
97  Real Cursor = 0;
98  Real Kerning = 0;
99 
100  unsigned char CurrChar = 0, LastChar = 0;
101  Glyph* CurrGlyph = 0;
102  Size.X = 0;
103  Size.Y = GlyphSet->LineHeight;
104 
105  for( Whole Index = 0 ; Index < Text.length() ; Index++ )
106  {
107  CurrChar = Text[Index];
108 
109  if(CurrChar == ' ')
110  {
111  LastChar = CurrChar;
112  Cursor += GlyphSet->SpaceLength;
113  continue;
114  }
115 
116  if( CurrChar < GlyphSet->RangeBegin || CurrChar > GlyphSet->RangeEnd )
117  {
118  LastChar = 0;
119  continue;
120  }
121 
122  CurrGlyph = GlyphSet->GetGlyph(CurrChar);
123  if(CurrGlyph == 0)
124  continue;
125  Kerning = CurrGlyph->GetKerning(LastChar);
126  if(Kerning == 0)
127  Kerning = GlyphSet->LetterSpacing;
128 
129  if(CharScaling == 1.0f)
130  Cursor += CurrGlyph->GlyphAdvance + Kerning;
131  else
132  Cursor += (CurrGlyph->GlyphAdvance + Kerning) * CharScaling;
133  LastChar = CurrChar;
134  }
135  Size.X = Cursor - Kerning;
136  }

251  void Caption::_Redraw()
252  {
253  if(Dirty == false)
254  return;
255  Rectangle::_Redraw();
256  if(!Visible)
257  {
258  Dirty = false;
259  return;
260  }
261 
262  Real Left = 0, Top = 0, Right = 0, Bottom = 0, CursorX = 0, CursorY = 0, Kerning = 0;
263  Real TexelOffsetX = ParentScreen->GetTexelOffsetX();
264  Real TexelOffsetY = ParentScreen->GetTexelOffsetY();
265  Vector2 KnownSize;
266  Glyph* CurrGlyph = 0;
267 
268  bool ClipLeft = false, ClipRight = false;
269  Real ClipLeftPos = 0, ClipRightPos = 0;
270 
271  if(HoriAlign == UI::Txt_Left)
272  {
273  CursorX = ActPosition.X + CursorOffset;
274 
275  ClipRight = true;
276  ClipRightPos = ActPosition.X + ActSize.X;
277  }
278  else if(HoriAlign == UI::Txt_Middle)
279  {
280  CalculateDrawSize(KnownSize);
281  CursorX = ActPosition.X + (ActSize.X * 0.5f) - (KnownSize.X * 0.5f);
282 
283  ClipLeft = true;
284  ClipLeftPos = ActPosition.X;
285  ClipRight = true;
286  ClipRightPos = ActPosition.X + ActSize.X;
287  }
288  else if(HoriAlign == UI::Txt_Right)
289  {
290  CalculateDrawSize(KnownSize);
291  CursorX = (ActPosition.X + ActSize.X) - (KnownSize.X + CursorOffset);
292 
293  ClipLeft = true;
294  ClipLeftPos = ActPosition.X;
295  }
296 
297  if(VertAlign == UI::Txt_Top)
298  CursorY = ActPosition.Y;
299  else if(VertAlign == UI::Txt_Center)
300  CursorY = ActPosition.Y + (ActSize.Y * 0.5) - ((GlyphSet->LineHeight * CharScaling) * 0.5);
301  else if(VertAlign == UI::Txt_Bottom)
302  CursorY = ActPosition.Y + ActSize.Y - (GlyphSet->LineHeight * CharScaling);
303 
304  unsigned char ThisChar = 0, LastChar = 0;
305  ClippedLeftIndex = String::npos;
306  ClippedRightIndex = String::npos;
307 
308  CursorX = MathTools::Floor( CursorX );
309  CursorY = MathTools::Floor( CursorY );
310 
311  for( Whole Index = 0 ; Index < Text.size() ; Index++ )
312  {
313  ThisChar = Text[Index];
314 
315  if( ThisChar == ' ' )
316  {
317  LastChar = ThisChar;
318  CursorX += GlyphSet->SpaceLength * CharScaling;
319  continue;
320  }
321 
322  if( ThisChar < GlyphSet->RangeBegin || ThisChar > GlyphSet->RangeEnd )
323  {
324  LastChar = 0;
325  continue;
326  }
327 
328  CurrGlyph = GlyphSet->GetGlyph(ThisChar);
329  if(CurrGlyph == 0)
330  continue;
331  Kerning = CurrGlyph->GetKerning(LastChar);
332  if(Kerning == 0)
333  Kerning = GlyphSet->LetterSpacing;
334 
335  Left = CursorX - TexelOffsetX;
336  Top = CursorY - TexelOffsetY + CurrGlyph->VerticalOffset;
337  Right = Left + (CurrGlyph->UVWidth * CharScaling) + TexelOffsetX;
338  Bottom = Top + (CurrGlyph->UVHeight * CharScaling) + TexelOffsetY;
339 
340  if(ClipLeft)
341  {
342  if (Left < ClipLeftPos)
343  {
344  if(ClippedLeftIndex == String::npos)
345  ClippedLeftIndex = Index;
346  CursorX += (CurrGlyph->GlyphAdvance + Kerning) * CharScaling;
347  LastChar = ThisChar;
348  continue;
349  }
350  }
351 
352  if(ClipRight)
353  {
354  if(Right > ClipRightPos)
355  {
356  if(ClippedRightIndex == String::npos)
357  ClippedRightIndex = Index;
358  CursorX += (CurrGlyph->GlyphAdvance + Kerning) * CharScaling;
359  LastChar = ThisChar;
360  continue;
361  }
362  }
363 
364  VertexData Temp;
365 
366  // Triangle A
367  PushVertex(RenderVertices, Temp, Left, Bottom, CurrGlyph->AtlasCoords[UI::QC_BottomLeft], TextColour,GlyphAtlas); // Left/Bottom 3
368  PushVertex(RenderVertices, Temp, Right, Top, CurrGlyph->AtlasCoords[UI::QC_TopRight], TextColour,GlyphAtlas); // Right/Top 1
369  PushVertex(RenderVertices, Temp, Left, Top, CurrGlyph->AtlasCoords[UI::QC_TopLeft], TextColour,GlyphAtlas); // Left/Top 0
370 
371  // Triangle B
372  PushVertex(RenderVertices, Temp, Left, Bottom, CurrGlyph->AtlasCoords[UI::QC_BottomLeft], TextColour,GlyphAtlas); // Left/Bottom 3
373  PushVertex(RenderVertices, Temp, Right, Bottom, CurrGlyph->AtlasCoords[UI::QC_BottomRight], TextColour,GlyphAtlas); // Right/Bottom 2
374  PushVertex(RenderVertices, Temp, Right, Top, CurrGlyph->AtlasCoords[UI::QC_TopRight], TextColour,GlyphAtlas); // Right/Top 1
375 
376  CursorX += (CurrGlyph->GlyphAdvance + Kerning) * CharScaling;
377  LastChar = ThisChar;
378  }
379  Dirty = false;
380  }
