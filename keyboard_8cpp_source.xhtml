61  void Keyboard::UpdateImpl(const MetaCodeContainer& DeltaCodes, MetaCodeContainer& GeneratedCodes)
62  {
63  for( Whole X = 0 ; X < DeltaCodes.size() ; ++X )
64  {
65  this->TransitioningIndexes.push_back( DeltaCodes[X].GetCode() );
66  this->Buttons.at( DeltaCodes[X].GetCode() ) = static_cast<Input::ButtonState>(DeltaCodes[X].GetMetaValue());
67  }
68  this->Sequences.Update(DeltaCodes,GeneratedCodes);
69  }

71  void Keyboard::VerifySequenceImpl(const MetaCodeContainer& Sequence) const
72  {
73  for( ConstMetaCodeIterator MCIt = Sequence.begin() ; MCIt != Sequence.end() ; ++MCIt )
74  {
75  if( !MCIt->IsKeyboardEvent() )
76  { MEZZ_EXCEPTION(Exception::PARAMETERS_EXCEPTION,"Non-Keyboard MetaCode detected when attempting to insert an Input Sequence into Keyboard input device."); }
77  }

80  void Keyboard::AddPressedButtons(MetaCodeContainer& GeneratedCodes) const
81  {
82  for( UInt32 Index = 0 ; Index < this->Buttons.size() ; ++Index )
83  {
84  if( this->Buttons.at(Index) == Input::BUTTON_DOWN )
85  GeneratedCodes.push_back( MetaCode(Input::BUTTON_DOWN,static_cast<Input::InputCode>(Input::KEY_FIRST + Index),GetDeviceIndex()) );
86  }
87  }
