91  void AddCharacter(Character* ToAdd, const Real& Scaling)
92  {
93  Real CharWidth = ToAdd->CharGlyph ? ToAdd->CharGlyph->GlyphAdvance + ToAdd->Kerning : ToAdd->CharSprite->SpriteSize.X;
94  CharWidth*=Scaling;
95  Pxlength += CharWidth;
96  Real CharHeight = ToAdd->CharGlyph ? ToAdd->CharGlyph->GlyphSize.Y : ToAdd->CharSprite->SpriteSize.Y;
97  CharHeight*=Scaling;
98  if(Pxheight < CharHeight)
99  Pxheight = CharHeight;
100  Characters.push_back(ToAdd);
101  }
