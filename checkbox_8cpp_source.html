58  CheckBox::CheckBox(ConstString& name, const RenderableRect& Rect, const Real& LineHeight, ConstString& LabelText, Screen* PScreen)
59  : Widget(name,PScreen),
60  Checked(false),
61  CheckLock(true)
62  {
63  Type = Widget::W_CheckBox;
64  const Vector2& WinDim = ParentScreen->GetViewportDimensions();
65  RenderableRect BoxRect, LabelRect;
66  if(Rect.Relative)
67  {
68  RelPosition = Rect.Position;
69  RelSize = Rect.Size;
70 
71  BoxRect.Position = Rect.Position;
72  BoxRect.Size.X = (Rect.Size.Y * WinDim.Y) / WinDim.X;
73  BoxRect.Size.Y = Rect.Size.Y;
74  BoxRect.Relative = Rect.Relative;
75  }else{
76  RelPosition = Rect.Position / WinDim;
77  RelSize = Rect.Size / WinDim;
78 
79  BoxRect.Position = Rect.Position;
80  BoxRect.Size.X = Rect.Size.Y;
81  BoxRect.Size.Y = Rect.Size.Y;
82  BoxRect.Relative = Rect.Relative;
83  }
84  LabelRect.Position.X = Rect.Position.X + BoxRect.Size.X;
85  LabelRect.Position.Y = Rect.Position.Y;
86  LabelRect.Size.X = Rect.Size.X - BoxRect.Size.X;
87  LabelRect.Size.Y = Rect.Size.Y;
88  LabelRect.Relative = Rect.Relative;
89 
90  Box = ParentScreen->CreateButton(Name+"CB",BoxRect);
91  Label = ParentScreen->CreateCaption(Name+"CM",LabelRect,LineHeight,LabelText);
92  AddSubRenderable(0,Box);
93  AddSubRenderable(1,Label);
94  }

148  void CheckBox::UpdateImpl(bool Force)
149  {
150  if( HoveredSubWidget && (Widget::W_Button == HoveredSubWidget->GetType()) )
151  {
152  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
153  if(Input::BUTTON_PRESSING == State)
154  {
155  CheckLock = false;
156  }
157  else if(Input::BUTTON_LIFTING && !CheckLock)
158  {
159  Checked = !Checked;
160  if(Checked)
161  {
162  SetSpriteSet(CheckedSet);
163  }else{
164  SetSpriteSet(UncheckedSet);
165  }
166  CheckLock = true;
167  }
168  }
169  }
