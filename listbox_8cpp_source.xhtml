183  void ListBox::DrawList()
184  {
185  if(Selections.empty())
186  return;
187  ScrollHideCheck();
188  VisibleSelections.clear();
189  Whole FirstCaption = 0;
190  if(MaxDisplay < Selections.size())
191  {
192  Real ToBeRounded = VertScroll->GetScrollerValue() * (Real)(Selections.size() - MaxDisplay);
193  FirstCaption = (Whole)(ToBeRounded + 0.5);
194  }
195  Vector2 SelectionPos = GetActualPosition();
196  Real ActualInc = SelectionTemplate.Size.Y;
197 
198  for( Whole w = 0 ; w < FirstCaption ; ++w )
199  {
200  Selections[w]->SetPosition(GetPosition());
201  Selections[w]->Hide();
202  SelectionSizeCheck(Selections[w]);
203  }
204  Whole Displayed = FirstCaption+MaxDisplay > Selections.size() ? Selections.size() : FirstCaption+MaxDisplay;
205  for( Whole x = FirstCaption ; x < Displayed ; ++x )
206  {
207  VisibleSelections.push_back(Selections[x]);
208  Selections[x]->SetVisible(this->IsVisible());
209  }
210  for( Whole y = Displayed ; y < Selections.size() ; ++y )
211  {
212  Selections[y]->SetPosition(GetPosition());
213  Selections[y]->Hide();
214  SelectionSizeCheck(Selections[y]);
215  }
216  for( Whole z = 0 ; z < VisibleSelections.size() ; ++z )
217  {
218  VisibleSelections[z]->SetActualPosition(SelectionPos);
219  SelectionPos.Y+=ActualInc;
220  SelectionSizeCheck(VisibleSelections[z]);
221  }
222  }

343  Caption* ListBox::AddSelection(ConstString& name, ConstString &Text, ConstString& BackgroundSprite)
344  {
345  SelectionsAdded++;
346  RenderableRect SelectionRect(RelPosition,SelectionTemplate.Size / ParentScreen->GetViewportDimensions(),true);
347  Caption* Select = ParentScreen->CreateCaption(name,SelectionRect,SelectionTemplate.GlyphIndex,Text);
348  if(!BackgroundSprite.empty())
349  Select->SetBackgroundSprite(BackgroundSprite);
350  if(SelectionTemplate.BackgroundColour != ColourValue(1.0,1.0,1.0,1.0))
351  Select->SetBackgroundColour(SelectionTemplate.BackgroundColour);
352  if(SelectionTemplate.CursorOffset != 0)
353  Select->SetCursorOffset(SelectionTemplate.CursorOffset);
354  if(SelectionTemplate.TextScale != 1)
355  Select->SetTextScale(SelectionTemplate.TextScale);
356  Select->SetTextColour(SelectionTemplate.TextColour);
357  Select->HorizontallyAlign(SelectionTemplate.HorizontalAlign);
358  Select->VerticallyAlign(SelectionTemplate.VerticalAlign);
359  Select->SetRenderPriority(SelectionTemplate.Priority);
360  Select->Hide();
361  Selections.push_back(Select);
362  AddSubRenderable(SelectionsAdded,Select);
363  ScrollerSizeCheck();
364  DrawList();
365  return Select;
366  }
