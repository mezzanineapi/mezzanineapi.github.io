359  bool ResourceManager::CreateDirectoryPath(const String& DirectoryPath)
360  {
361  bool Result = true;
362  StringVector FolderNames;
363  CountedPtr<StringVector> FolderVec = StringTools::Split(DirectoryPath,"/\\");
364  size_t StartIndex = 0;
365  String PathAttempt;
366  char SysSlash = GetDirectorySeparator();
367  #ifdef WINDOWS
368  // For windows and windows like machines, see if the first entry is a drive, because attempting to make a drive is silly.
369  if(FolderVec->at(0).find(':') != String::npos)
370  {
371  PathAttempt.append( FolderVec->at(0) );
372  PathAttempt.append( 1, SysSlash );
373  StartIndex++;
374  }
375  #else
376  PathAttempt.append( 1, SysSlash );
377  #endif
378  for( size_t VecIndex = StartIndex ; VecIndex < FolderVec->size() ; ++VecIndex )
379  {
380  PathAttempt.append( FolderVec->at(VecIndex) );
381  PathAttempt.append( 1, SysSlash );
382  Result = this->CreateDirectory( PathAttempt );
383  }
384  return Result;
385  }
