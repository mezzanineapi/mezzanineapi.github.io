103  void DeviceUpdateWorkUnit::DoWork(Threading::DefaultThreadSpecificStorage::Type& CurrentThreadStorage)
104  {
105  // Set up our containers for the update
106  this->TargetManager->InputDeltas.clear();
107  std::vector< Input::MetaCode > MouseCodes;
108  std::vector< Input::MetaCode > KeyboardCodes;
109  std::vector< std::vector< Input::MetaCode > > ControllerCodes;
110  ControllerCodes.resize( this->TargetManager->GetNumControllers() );
111  // And finally our container for our generated codes
112  std::vector< Input::MetaCode > GeneratedCodes;
113 
114  // Get the user input events for processing
115  std::list<EventUserInput*>* UserInput = EventManager::GetSingletonPtr()->GetAllUserInputEvents();
116  EventManager::GetSingletonPtr()->RemoveAllSpecificEvents(EventBase::UserInput);
117 
118  // Process the aquired user input events
119  while( !UserInput->empty() )
120  {
121  EventUserInput* CurrEvent = UserInput->front();
122  for( Whole X = 0 ; X < CurrEvent->size() ; ++X )
123  {
124  Input::MetaCode& CurrCode = CurrEvent->at(X);
125  this->TargetManager->InputDeltas.push_back(CurrCode);
126  if( Input::MOUSE_FIRST <= CurrCode.GetCode() && Input::MOUSE_LAST >= CurrCode.GetCode() ){
127  MouseCodes.push_back( CurrCode );
128  }else if( Input::KEY_FIRST <= CurrCode.GetCode() && Input::KEY_LAST >= CurrCode.GetCode() ){
129  KeyboardCodes.push_back( CurrCode );
130  }else if( Input::CONTROLLER_FIRST <= CurrCode.GetCode() && Input::CONTROLLER_LAST >= CurrCode.GetCode() ){
131  ControllerCodes[ CurrCode.GetDeviceIndex() ].push_back( CurrCode );
132  }
133  }
134 
135  delete CurrEvent;
136  UserInput->pop_front();
137  }
138  delete UserInput;
139  UserInput = NULL;
140 
141  // Update all of our devices with the processed/saved data
142  this->TargetManager->SystemMouse->_Update(MouseCodes,GeneratedCodes);
143  this->TargetManager->SystemKeyboard->_Update(KeyboardCodes,GeneratedCodes);
144  for( Whole X = 0 ; X < this->TargetManager->GetNumControllers() ; ++X )
145  {
146  this->TargetManager->IMID->Controllers.at(X).first->_Update( ControllerCodes.at(X) , GeneratedCodes );
147  }
148  // Do sub-system wide sequence checks if we've done anything
149  if( !this->TargetManager->InputDeltas.empty() )
150  this->TargetManager->Sequences.Update(this->TargetManager->InputDeltas,GeneratedCodes);
151  // Update our delta's if there is anything to update
152  if( !GeneratedCodes.empty() )
153  this->TargetManager->InputDeltas.insert(this->TargetManager->InputDeltas.end(),GeneratedCodes.begin(),GeneratedCodes.end());
154  }

182  this->SystemMouse = new Mouse();
183  this->SystemKeyboard = new Keyboard();

254  UInt16 InputManager::DetectControllers()
255  {
256  UInt16 Count;
257  for( Count = 0 ; Count < SDL_NumJoysticks() ; ++Count )
258  {
259  SDL_Joystick* InternalControl = SDL_JoystickOpen(Count);
260  Input::Controller* NewController = new Input::Controller( InternalControl, Count );
261  this->IMID->Controllers.push_back( InternalControlPair(NewController,InternalControl) );
262  }
263  return Count;
264  }

266  void InputManager::ReleaseAllControllers()
267  {
268  if( this->IMID->Controllers.empty() )
269  return;
270 
271  for( InternalControlIterator ContIt = this->IMID->Controllers.begin() ; ContIt != this->IMID->Controllers.end() ; ++ContIt )
272  {
273  delete (*ContIt).first;
274  SDL_JoystickClose( (SDL_Joystick*)(*ContIt).second );
275  }
276  this->IMID->Controllers.clear();
277  }
