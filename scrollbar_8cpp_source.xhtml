122  void Scrollbar::CreateHorizontalScrollbar(const RenderableRect& Rect)
123  {
124  Horizontal = true;
125  //Get the size for all items involved
126  RenderableRect ScrollBackRect, LeftButtonRect, RightButtonRect;
127  ScrollBackRect.Relative = Rect.Relative;
128  LeftButtonRect.Relative = Rect.Relative;
129  RightButtonRect.Relative = Rect.Relative;
130  if(UI::SB_NoButtons==ScrollStyle)
131  {
132  ScrollBackRect.Size = Rect.Size;
133  }else{
134  if(Rect.Relative)
135  {
136  const Vector2& WinDim = ParentScreen->GetViewportDimensions();
137  Vector2 ARectSize(Rect.Size * WinDim);
138 
139  ScrollBackRect.Size.X = Rect.Size.X - ((ARectSize.Y * 2) / WinDim.X);
140  ScrollBackRect.Size.Y = Rect.Size.Y;
141  LeftButtonRect.Size.X = ARectSize.Y / WinDim.X;
142  LeftButtonRect.Size.Y = Rect.Size.Y;
143  RightButtonRect.Size.X = ARectSize.Y / WinDim.X;
144  RightButtonRect.Size.Y = Rect.Size.Y;
145  }else{
146  ScrollBackRect.Size.X = Rect.Size.X - (Rect.Size.Y * 2);
147  ScrollBackRect.Size.Y = Rect.Size.Y;
148  LeftButtonRect.Size.X = Rect.Size.Y;
149  LeftButtonRect.Size.Y = Rect.Size.Y;
150  RightButtonRect.Size.X = Rect.Size.Y;
151  RightButtonRect.Size.Y = Rect.Size.Y;
152  }
153  }
154  //Get the position for all items involved and configure their offsets
155  if(UI::SB_Separate==ScrollStyle)
156  {
157  ScrollBackRect.Position.X = Rect.Position.X + LeftButtonRect.Size.X;
158  ScrollBackRect.Position.Y = Rect.Position.Y;
159  LeftButtonRect.Position = Rect.Position;
160  RightButtonRect.Position.X = Rect.Position.X + LeftButtonRect.Size.X + ScrollBackRect.Size.X;
161  RightButtonRect.Position.Y = Rect.Position.Y;
162  }
163  else if(UI::SB_TogetherDownRight==ScrollStyle)
164  {
165  ScrollBackRect.Position = Rect.Position;
166  LeftButtonRect.Position.X = Rect.Position.X + ScrollBackRect.Size.X;
167  LeftButtonRect.Position.Y = Rect.Position.Y;
168  RightButtonRect.Position.X = Rect.Position.X + ScrollBackRect.Size.X + LeftButtonRect.Size.X;
169  RightButtonRect.Position.Y = Rect.Position.Y;
170  }
171  else if(UI::SB_TogetherUpLeft==ScrollStyle)
172  {
173  ScrollBackRect.Position.X = Rect.Position.X + LeftButtonRect.Size.X + RightButtonRect.Size.X;
174  ScrollBackRect.Position.Y = Rect.Position.Y;
175  LeftButtonRect.Position = Rect.Position;
176  RightButtonRect.Position.X = Rect.Position.X + LeftButtonRect.Size.X;
177  RightButtonRect.Position.Y = Rect.Position.Y;
178  }
179  else
180  {
181  ScrollBackRect.Position = Rect.Position;
182  }
183  //Now create the objects, since we have all the position and size data we need
184  ScrollBack = ParentScreen->CreateRectangle(ScrollBackRect);
185  Scroller = ParentScreen->CreateButton(Name+"SC",ScrollBackRect);
186  if(LeftButtonRect.Size.X > 0 && LeftButtonRect.Size.Y > 0)
187  {
188  UpLeftButton = ParentScreen->CreateButton(Name+"LB",LeftButtonRect);
189  }
190  if(RightButtonRect.Size.X > 0 && RightButtonRect.Size.Y > 0)
191  {
192  DownRightButton = ParentScreen->CreateButton(Name+"RB",RightButtonRect);
193  }
194  AddSubRenderable(0,ScrollBack);
195  AddSubRenderable(1,Scroller);
196  AddSubRenderable(2,UpLeftButton);
197  AddSubRenderable(3,DownRightButton);
198  CalculateScrollLimits();
199  }

305  void Scrollbar::CalculateOffsets(const Vector2& Size)
306  {
307  if(Horizontal)
308  {
309  if(UI::SB_Separate==ScrollStyle)
310  {
311  ScrollBackOffset = Vector2(Size.Y,0);
312  UpLeftButtonOffset = Vector2(0,0);
313  DownRightButtonOffset = Vector2((Size.X - (Size.Y * 2)) + Size.Y,0);
314  }
315  else if(UI::SB_TogetherDownRight==ScrollStyle)
316  {
317  ScrollBackOffset = Vector2(0,0);
318  UpLeftButtonOffset = Vector2(Size.X - (Size.Y * 2),0);
319  DownRightButtonOffset = Vector2((Size.X - (Size.Y * 2)) + Size.Y,0);
320  }
321  else if(UI::SB_TogetherUpLeft==ScrollStyle)
322  {
323  ScrollBackOffset = Vector2(Size.Y * 2,0);
324  UpLeftButtonOffset = Vector2(0,0);
325  DownRightButtonOffset = Vector2(Size.Y,0);
326  }
327  }else{
328  if(UI::SB_Separate==ScrollStyle)
329  {
330  ScrollBackOffset = Vector2(0,Size.X);
331  UpLeftButtonOffset = Vector2(0,0);
332  DownRightButtonOffset = Vector2(0,(Size.Y - (Size.X * 2)) + Size.X);
333  }
334  else if(UI::SB_TogetherDownRight==ScrollStyle)
335  {
336  ScrollBackOffset = Vector2(0,0);
337  UpLeftButtonOffset = Vector2(0,Size.Y - (Size.X * 2));
338  DownRightButtonOffset = Vector2(0,(Size.Y - (Size.X * 2)) + Size.X);
339  }
340  else if(UI::SB_TogetherUpLeft==ScrollStyle)
341  {
342  ScrollBackOffset = Vector2(0,Size.X * 2);
343  UpLeftButtonOffset = Vector2(0,0);
344  DownRightButtonOffset = Vector2(0,Size.X);
345  }
346  }

361  void Scrollbar::CalculateScrollValue()
362  {
363  Real Pos = Horizontal ? Scroller->GetActualPosition().X : Scroller->GetActualPosition().Y;
364  Real Size = Horizontal ? Scroller->GetActualSize().X : Scroller->GetActualSize().Y;
365  Real RelPos = Pos - ScrollerLowerLimit;
366  Real RelLimit = (ScrollerUpperLimit - ScrollerLowerLimit) - Size;
367  ScrollerValue = RelPos / RelLimit;
368  }

370  void Scrollbar::SetToWithinLimits(Real& Coord)
371  {
372  Real SizeOffset = Horizontal ? Scroller->GetActualSize().X : Scroller->GetActualSize().Y;
373  if(Coord < ScrollerLowerLimit)
374  {
375  Coord = ScrollerLowerLimit;
376  }
377  else if(Coord + SizeOffset > ScrollerUpperLimit)
378  {
379  Coord = ScrollerUpperLimit - SizeOffset;
380  }

383  void Scrollbar::SetLocation(const Vector2& Position)
384  {
385  ScrollBack->SetActualPosition(Position + ScrollBackOffset);
386  CalculateScrollLimits();
387  Vector2 ScrollVal(0,0);
388  Real SizeOffset = Horizontal ? Scroller->GetActualSize().X : Scroller->GetActualSize().Y;
389  if(Horizontal) ScrollVal.X = ((ScrollerUpperLimit - ScrollerLowerLimit) - SizeOffset) * ScrollerValue;
390  else ScrollVal.Y = ((ScrollerUpperLimit - ScrollerLowerLimit) - SizeOffset) * ScrollerValue;
391  Scroller->SetActualPosition(Position + ScrollBackOffset + ScrollVal);

394  UpLeftButton->SetActualPosition(Position + UpLeftButtonOffset);
395  }
396  if(DownRightButton)
397  {
398  DownRightButton->SetActualPosition(Position + DownRightButtonOffset);
399  }

437  void Scrollbar::ButtonScroll(bool UpLeft)
438  {
439  if((!UpLeftLock && HoveredSubWidget == UpLeftButton) || (!DownRightLock && HoveredSubWidget == DownRightButton))
440  {
441  Vector2 Loc = Scroller->GetActualPosition();
442  Real Scroll = ScrollBack->GetActualSize().X * IncrementDistance;
443  if(UpLeft)
444  {
445  Scroll = -Scroll;
446  }
447  if(Horizontal)
448  {
449  Loc.X+=Scroll;
450  SetToWithinLimits(Loc.X);
451  }else{
452  Loc.Y+=Scroll;
453  SetToWithinLimits(Loc.Y);
454  }
455  Scroller->SetActualPosition(Loc);
456  }
457  }

459  void Scrollbar::MouseScroll(Vector2& Scroll)
460  {
461  if(!ScrollerLock)
462  {
463  Vector2 Loc = Scroller->GetActualPosition();
464  if(Horizontal)
465  {
466  Scroll.Y = Loc.Y;
467  Scroll.X = Loc.X + Scroll.X;
468  SetToWithinLimits(Scroll.X);
469  }else{
470  Scroll.X = Loc.X;
471  Scroll.Y = Loc.Y + Scroll.Y;
472  SetToWithinLimits(Scroll.Y);
473  }
474  Scroller->SetActualPosition(Scroll);
475  }

478  void Scrollbar::ScrollBackScroll()
479  {
480  if(!ScrollBackLock)
481  {
482  Vector2 MousePos = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetViewportPosition();
483  Vector2 ScPos = Scroller->GetActualPosition();
484  Vector2 ScSize = Scroller->GetActualSize();
485  if(Horizontal)
486  {
487  if(MousePos.X < ScPos.X)//left
488  {
489  ScPos.X = ScPos.X - ScSize.X <= ScrollerLowerLimit ? ScrollerLowerLimit : ScPos.X - ScSize.X ;
490  Scroller->SetActualPosition(ScPos);
491  }
492  else if(MousePos.X > ScPos.X + ScSize.X)//right
493  {
494  ScPos.X = ScPos.X + (ScSize.X * 2) >= ScrollerUpperLimit ? ScrollerUpperLimit - ScSize.X : ScPos.X + (ScSize.X * 2) ;
495  Scroller->SetActualPosition(ScPos);
496  }
497  }else{
498  if(MousePos.Y < ScPos.Y)//above
499  {
500  ScPos.Y = ScPos.Y - ScSize.Y <= ScrollerLowerLimit ? ScrollerLowerLimit : ScPos.Y - ScSize.Y ;
501  Scroller->SetActualPosition(ScPos);
502  }
503  else if(MousePos.Y > ScPos.Y + ScSize.Y)//below
504  {
505  ScPos.Y = ScPos.Y + (ScSize.Y * 2) >= ScrollerUpperLimit ? ScrollerUpperLimit - ScSize.Y : ScPos.Y + ScSize.Y ;
506  Scroller->SetActualPosition(ScPos);
507  }
508  }
509  }

512  void Scrollbar::UpdateImpl(bool Force)
513  {
514  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
515  if( HoveredSubWidget && (Widget::W_Button == HoveredSubWidget->GetType()) )
516  {
517  if(Input::BUTTON_PRESSING == State || Input::BUTTON_DOWN == State)
518  {
519  if(HoveredSubWidget == Scroller)
520  {
521  if(Input::BUTTON_PRESSING == State)
522  {
523  ScrollerLock = false;
524  }
525  else if(Input::BUTTON_DOWN == State)
526  {
527  Vector2 Offset = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetMouseDelta();
528  MouseScroll(Offset);
529  }
530  }
531  else if(HoveredSubWidget == UpLeftButton)
532  {
533  if(Input::BUTTON_PRESSING == State)
534  {
535  UpLeftLock = false;
536  }
537  else if(Input::BUTTON_DOWN == State)
538  {
539  ButtonScroll(true);
540  }
541  }
542  else if(HoveredSubWidget == DownRightButton)
543  {
544  if(Input::BUTTON_PRESSING == State)
545  {
546  DownRightLock = false;
547  }
548  else if(Input::BUTTON_DOWN == State)
549  {
550  ButtonScroll(false);
551  }
552  }
553  }
554  }
555  else if(HoveredBack)
556  {
557  if(Input::BUTTON_PRESSING == State)
558  {
559  ScrollBackLock = false;
560  ScrollBackScroll();
561  }
562  else if(Input::BUTTON_DOWN == State)
563  {
564  if(ScrollBackLock && !ScrollerLock)
565  {
566  Vector2 Offset = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetMouseDelta();
567  MouseScroll(Offset);
568  }else{
569  ScrollBackScroll();
570  }
571  }
572  }
573  else if(Force && Input::BUTTON_DOWN == State)
574  {
575  Vector2 Offset = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetMouseDelta();
576  MouseScroll(Offset);
577  }
578  else if(Input::BUTTON_LIFTING == State)
579  {
580  ScrollerLock = true;
581  ScrollBackLock = true;
582  UpLeftLock = true;
583  DownRightLock = true;
584  }
585  CalculateScrollValue();
586  }

627  void Scrollbar::SetScrollerValue(const Real& Value)
628  {
629  Vector2 ScrollerPos = Scroller->GetActualPosition();
630  Vector2 NewPos;
631  if(Horizontal)
632  {

638  }

653  void Scrollbar::SetScrollerSize(const Real& Size)
654  {
655  if(Size > 1 || Size < 0)
656  return;
657  Vector2 NewSize;
658  Vector2 SB = ScrollBack->GetActualSize();
