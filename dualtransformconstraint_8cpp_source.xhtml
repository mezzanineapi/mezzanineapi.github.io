55  void DualTransformConstraint::ProtoSerialize(XML::Node& CurrentRoot) const
56  {
57  XML::Node DualTransformConstraintNode = CurrentRoot.AppendChild(SerializableName()); // The base node all the base constraint stuff will go in
58  if (!DualTransformConstraintNode)
59  { SerializeError("Create DualTransformConstraintNode", SerializableName()); }
60 
61  XML::Attribute Version = DualTransformConstraintNode.AppendAttribute("Version");
62  if (Version)
63  {
64  Version.SetValue(1);
65  }else{
66  SerializeError("Create Attributes on DualTransformConstraintNode", SerializableName());
67  }
68 
69  XML::Node ActorANode = DualTransformConstraintNode.AppendChild("ActorA"); // Get everything we need about ActorA
70  if (!ActorANode)
71  { SerializeError("Create ActorANode", SerializableName()); }
72  this->GetPivotATransform().ProtoSerialize(ActorANode);
73 
74  XML::Node ActorBNode = DualTransformConstraintNode.AppendChild("ActorB"); // Get everything we need about ActorB
75  if (!ActorBNode)
76  { SerializeError("Create ActorBNode", SerializableName()); }
77  this->GetPivotBTransform().ProtoSerialize(ActorBNode);
78 
79  this->Constraint::ProtoSerialize(DualTransformConstraintNode);

82  // DeSerializable
83  void DualTransformConstraint::ProtoDeSerialize(const XML::Node& OneNode)
84  {
85  if ( Mezzanine::String(OneNode.Name())==this->DualTransformConstraint::SerializableName() )
86  {
87  if(OneNode.GetAttribute("Version").AsInt() == 1)
88  {
89  XML::Node ConstraintNode = OneNode.GetChild("Constraint");
90  if(!ConstraintNode)
91  { DeSerializeError("locate Constraint node",SerializableName()); }
92  this->Constraint::ProtoDeSerialize(ConstraintNode);
93 
94  XML::Node TransformA = OneNode.GetChild("ActorA").GetFirstChild();
95  if (!TransformA)
96  { DeSerializeError("locate transform for ActorA",SerializableName()); }
97  Transform temp;
98  temp.ProtoDeSerialize(TransformA);
99  this->SetPivotATransform(temp);
100 
101  XML::Node TransformB = OneNode.GetChild("ActorB").GetFirstChild();
102  if (!TransformB)
103  { DeSerializeError("locate transform for ActorB",SerializableName()); }
104  temp.ProtoDeSerialize(TransformB);
105  this->SetPivotBTransform(temp);
106  }else{
107  DeSerializeError("find usable serialization version",SerializableName());
108  }
109  }else{
110  DeSerializeError(String("find correct class to deserialize, found a ")+OneNode.Name(),SerializableName());
111  }
112  }
