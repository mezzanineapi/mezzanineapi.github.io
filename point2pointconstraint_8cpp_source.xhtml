144  void Point2PointConstraint::ProtoSerialize(XML::Node& CurrentRoot) const
145  {
146  XML::Node P2PNode = CurrentRoot.AppendChild(SerializableName()); // The base node all the base constraint stuff will go in
147  if (!P2PNode)
148  { SerializeError("Create P2PNode", SerializableName()); }
149 
150  XML::Attribute VerAttr = P2PNode.AppendAttribute("Version");
151  XML::Attribute TauAttr = P2PNode.AppendAttribute("Tau");
152  XML::Attribute ClaAttr = P2PNode.AppendAttribute("ImpulseClamping");
153  XML::Attribute DamAttr = P2PNode.AppendAttribute("Damping");
154 
155  if( VerAttr && TauAttr && ClaAttr && DamAttr )
156  {
157  VerAttr.SetValue(1);
158  TauAttr.SetValue(this->GetTAU());
159  ClaAttr.SetValue(this->GetImpulseClamping());
160  DamAttr.SetValue(this->GetDamping());
161  }else{
162  SerializeError("Create P2PNode Attributes", SerializableName());
163  }
164 
165  XML::Node ActorANode = P2PNode.AppendChild("ActorA");
166  if (!ActorANode)
167  { SerializeError("Create ActorANode", SerializableName()); }
168  this->GetPivotALocation().ProtoSerialize(ActorANode);
169 
170  XML::Node ActorBNode = P2PNode.AppendChild("ActorB");
171  if (!ActorBNode)
172  { SerializeError("Create ActorBNode", SerializableName()); }
173  this->GetPivotBLocation().ProtoSerialize(ActorBNode);
174 
175  this->Constraint::ProtoSerialize(P2PNode);
176  }

178  void Point2PointConstraint::ProtoDeSerialize(const XML::Node& OneNode)
179  {
