91  void DropDownList::ConstructDropDownList(const RenderableRect& Rect, const Whole& Glyph, const UI::ScrollbarStyle& ScrollStyle)
92  {
93  Type = Widget::W_DropDownList;
94  RenderableRect SelectionRect, ListToggleRect, SelectionListRect;
95  Real ScrollbarWidth;
96  const Vector2& WinDim = ParentScreen->GetViewportDimensions();
97  if(Rect.Relative)
98  {
99  RelPosition = Rect.Position;
100  RelSize = Rect.Size;
101 
102  SelectionRect.Position = Rect.Position;
103  SelectionRect.Size.X = Rect.Size.X - ((Rect.Size.Y * WinDim.Y) / WinDim.X);
104  SelectionRect.Size.Y = Rect.Size.Y;
105  SelectionRect.Relative = Rect.Relative;
106 
107  ListToggleRect.Position.X = Rect.Position.X + SelectionRect.Size.X;
108  ListToggleRect.Position.Y = Rect.Position.Y;
109  ListToggleRect.Size.X = (Rect.Size.Y * WinDim.Y) / WinDim.X;
110  ListToggleRect.Size.Y = Rect.Size.Y;
111  ListToggleRect.Relative = Rect.Relative;
112 
113  ScrollbarWidth = (Rect.Size.Y * WinDim.Y) / WinDim.X;
114  }else{
115  RelPosition = Rect.Position / WinDim;
116  RelSize = Rect.Size / WinDim;
117 
118  SelectionRect.Position = Rect.Position;
119  SelectionRect.Size.X = Rect.Size.X - Rect.Size.Y;
120  SelectionRect.Size.Y = Rect.Size.Y;
121  SelectionRect.Relative = Rect.Relative;
122 
123  ListToggleRect.Position.X = Rect.Position.X + SelectionRect.Size.X;
124  ListToggleRect.Position.Y = Rect.Position.Y;
125  ListToggleRect.Size.X = Rect.Size.Y;
126  ListToggleRect.Size.Y = Rect.Size.Y;
127  ListToggleRect.Relative = Rect.Relative;
128 
129  ScrollbarWidth = Rect.Size.Y;
130  }
131  SelectionListRect.Position.X = Rect.Position.X;
132  SelectionListRect.Position.Y = Rect.Position.Y + SelectionRect.Size.Y;
133  SelectionListRect.Size = Rect.Size;
134  SelectionListRect.Relative = Rect.Relative;
135 
136  Selection = ParentScreen->CreateCaption(Name+"Select",SelectionRect,Glyph,"");
137  ListToggle = ParentScreen->CreateButton(Name+"Toggle",ListToggleRect);
138  SelectionList = ParentScreen->CreateListBox(Name+"List",SelectionListRect,ScrollStyle);
139 
140  AddSubRenderable(0,Selection);
141  AddSubRenderable(1,ListToggle);
142  AddSubRenderable(2,SelectionList);
143 
144  SelectionList->SetTemplateGlyphIndex(Glyph);
145  SelectionList->SetTemplateRenderPriority(UI::RP_High);
146  SelectionList->Hide();
147  SelectionList->GetBoxBack()->SetRenderPriority(UI::RP_High);
148  SelectionList->GetVertScroll()->GetScrollBack()->SetRenderPriority(UI::RP_High);
149  SelectionList->GetVertScroll()->GetScroller()->SetRenderPriority(UI::RP_High);
150  SelectionList->GetVertScroll()->GetUpLeftButton()->SetRenderPriority(UI::RP_High);
151  SelectionList->GetVertScroll()->GetDownRightButton()->SetRenderPriority(UI::RP_High);
152  }

154  void DropDownList::UpdateImpl(bool Force)
155  {
156  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
157  if(HoveredSubWidget == ListToggle)
158  {
159  if(Input::BUTTON_PRESSING == State)
160  {
161  if(!ToggleActivated)
162  {
163  SelectionList->Show();
164  ToggleActivated = true;
165  }else{
166  SelectionList->Hide();
167  ToggleActivated = false;
168  }
169  }
170  }
171  else if(HoveredSubWidget == SelectionList)
172  {
173  if(Input::BUTTON_PRESSING == State && !SelectionList->GetHoveredSubWidget())
174  {
175  Selection->SetText(SelectionList->GetSelected()->GetText());
176  SelectionList->Hide();
177  ToggleActivated = false;
178  }
179  }
180  }
