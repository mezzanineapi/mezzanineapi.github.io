69  void RigidProxy::CreateRigidObject(const Real& Mass)
70  {
71  this->PhysicsRigidBody = new btRigidBody(Mass, NULL/* MotionState */, NULL/* CollisionShape */);
72  PhysicsRigidBody->setUserPointer(this);
73  if(0.0 == Mass) {
74  PhysicsRigidBody->setCollisionFlags(btCollisionObject::CF_STATIC_OBJECT);
75  }else{
76  PhysicsRigidBody->setCollisionFlags(PhysicsRigidBody->getCollisionFlags() & (~btCollisionObject::CF_STATIC_OBJECT));

111  void RigidProxy::SetCollisionShape(CollisionShape* Shape)
112  {
113  if(CollisionShape::ST_StaticTriMesh != Shape->GetType())
114  {
115  btScalar Mass = this->PhysicsRigidBody->getInvMass();
116  if(0 != Mass)
117  Mass = 1 / Mass;
118  btVector3 Inertia(0,0,0);
119  Shape->_GetInternalShape()->calculateLocalInertia(Mass,Inertia);
120  this->PhysicsRigidBody->setMassProps(Mass,Inertia);
121  this->PhysicsProxy::SetCollisionShape(Shape);
122  this->PhysicsRigidBody->updateInertiaTensor();
123  }else{
124  this->PhysicsProxy::SetCollisionShape(Shape);
125  }
126  CollisionShapeManager::GetSingletonPtr()->StoreShape(Shape);
127  }
