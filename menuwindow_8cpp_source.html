169  MenuWindow* MenuWindow::CreateChildMenuWindow(ConstString& Name, const RenderableRect& Rect, Button* Accessor)
170  {
171  std::pair<Button*,MenuWindow*>* AccessorPair = NULL;
172  for( std::vector<std::pair<Button*,MenuWindow*> >::iterator it = ChildWindows.begin() ; it != ChildWindows.end() ; it++ )
173  {
174  if(Accessor == (*it).first)
175  AccessorPair = &(*it);
176  }
177  if(NULL==AccessorPair)
178  return 0;
179  MenuWindow* MenWin = new MenuWindow(Name,Rect,MasterMenu,ParentScreen);
180  MenWin->ParentWindow = this;
181  AccessorPair->second = MenWin;
182  MenWin->Hide();
183  return MenWin;
184  }
