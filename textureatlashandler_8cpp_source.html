65  void TextureAtlasHandler::LoadAtlasFromFile(const String& Name, const String& Group)
66  {
67  // Kinda hackish until we get the resource system running the way we want.
68  /// @todo Fix this when we have a working resource system of our own.
69  Resource::FileStream AtlasStream( Name, ResourceManager::GetSingletonPtr()->GetAssetPath(Name,Group) );
70  XML::Document AtlasDoc;
71  AtlasDoc.Load(AtlasStream);
72 
73  XML::Node RootNode = AtlasDoc.GetChild("Atlases");
74  if( !RootNode.Empty() )
75  {
76  for( XML::NodeIterator AtlasIt = RootNode.begin() ; AtlasIt != RootNode.end() ; ++AtlasIt )
77  {
78  // Parse the Atlas
79  TextureAtlas* NewAtlas = new TextureAtlas( (*AtlasIt) );
80  // Verify we don't already have one of the same name
81  AtlasIterator AtIt = Atlases.find( NewAtlas->GetName() );
82  if( AtIt != Atlases.end() )
83  {
84  MEZZ_EXCEPTION(Exception::II_DUPLICATE_IDENTITY_EXCEPTION,"Texture Atlas with the name \"" + NewAtlas->GetName() + "\" already exists.");
85  }
86  // Add the unique Atlas
87  Atlases[NewAtlas->GetName()] = NewAtlas;
88  }
89  }else{
90  MEZZ_EXCEPTION(Exception::INVALID_STATE_EXCEPTION,"Mezzanine Texture Atlas file \"" + Name + "\"does not contain expected \"Atlases\" root node. File is not valid and cannot be parsed.");
91  }
92  }

94  UI::TextureAtlas* TextureAtlasHandler::GetAtlas(const String& AtlasName)
95  {
96  AtlasIterator It = Atlases.find(AtlasName);
97  if(It != Atlases.end()) return (*It).second;
98  else { MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,"Unabled to find Texture Atlas named \"" + AtlasName + "\".") };
99  }
