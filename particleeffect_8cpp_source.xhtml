347 Mezzanine::XML::Node& operator >> (const Mezzanine::XML::Node& OneNode, Mezzanine::ParticleEffect& Ev)
348 {
349  if ( Mezzanine::String(OneNode.Name())==Mezzanine::String("ParticleEffect") )
350  {
351  if(OneNode.GetAttribute("Version").AsInt() == 1)
352  {
353  if(OneNode.GetAttribute("InWorld").AsBool())
354  {
355  Ev.AddToWorld();
356  }else{
357  Ev.RemoveFromWorld();
358  }
359 
360  Mezzanine::WorldNode * AttachPtr = Mezzanine::Entresol::GetSingletonPtr()->GetSceneManager()->GetNode( OneNode.GetAttribute("AttachedTo").AsString() );
361  if (AttachPtr)
362  { AttachPtr->AttachObject(&Ev); }
363 
364  Mezzanine::Quaternion TempQuat(0,0,0,0);
365  Mezzanine::Vector3 TempVec(0,0,0);
366 
367  for(Mezzanine::XML::Node Child = OneNode.GetFirstChild(); Child!=0; Child = Child.GetNextSibling())
368  {
369  Mezzanine::String Name(Child.Name());
370  switch(Name[0])
371  {
372  case 'O': //Orientation
373  if(Name==Mezzanine::String("Orientation"))
374  {
375  Child.GetFirstChild() >> TempQuat;
376  Ev.SetOrientation(TempQuat);
377  }else{
378  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for WorldNode: Includes unknown Element D-\"" + Name + "\".");
379  }
380  break;
381  case 'L': //Location
382  if(Name==Mezzanine::String("Location"))
383  {
384  Child.GetFirstChild() >> TempVec;
385  Ev.SetLocation(TempVec);
386  }else{
387  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for ParticleEffect: Includes unknown Element L-\"" + Name + "\".");
388  }
389  break;
390  default:
391  { MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for ParticleEffect: Includes unknown Element default-\"" + Name + "\"."); }
392  break;
393  }
394  }
395 
396  }else{
397  MEZZ_EXCEPTION(Mezzanine::Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for ParticleEffect: Not Version 1.");
398  }
399  }else{
400  MEZZ_EXCEPTION(Mezzanine::Exception::II_IDENTITY_INVALID_EXCEPTION,"Attempting to deserialize a ParticleEffect, found a " + Mezzanine::String(OneNode.Name()));
401  }
