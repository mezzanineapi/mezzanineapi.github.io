98  void Plane::Define(const Vector3& First, const Vector3& Second, const Vector3& Third)
99  {
100  Vector3 Edge1 = Second - First;
101  Vector3 Edge2 = Third - First;
102  this->Normal = Edge1.CrossProduct(Edge2);
103  this->Normal.Normalize();
104  this->Distance = -(this->Normal.DotProduct(First));
105  }

119  Plane::Side Plane::GetSide(const Vector3& Center, const Vector3& HalfSize) const
120  {
121  Real CenterDist = this->GetDistance(Center);
122  Real MaxDist = MathTools::Fabs( this->Normal.DotProduct(HalfSize) );
123  if( CenterDist < -MaxDist ) {

159  void Plane::ProtoSerialize(XML::Node& ParentNode) const
160  {
161  XML::Node SelfRoot = ParentNode.AppendChild( Plane::GetSerializableName() );
162 
163  if( SelfRoot.AppendAttribute("Version").SetValue("1") &&
164  SelfRoot.AppendAttribute("Distance").SetValue( this->Distance ) )
165  {
166  XML::Node CenterNode = SelfRoot.AppendChild("Normal");
167  this->Normal.ProtoSerialize( CenterNode );
168 
169  return;
170  }else{
171  SerializeError("Create XML Attribute Values",Plane::GetSerializableName(),true);
172  }
173  }

175  void Plane::ProtoDeSerialize(const XML::Node& SelfRoot)
176  {
177  XML::Attribute CurrAttrib;
178 
179  if( String(SelfRoot.Name()) == Plane::GetSerializableName() ) {
180  if(SelfRoot.GetAttribute("Version").AsInt() == 1) {
181  CurrAttrib = SelfRoot.GetAttribute("Distance");
182  if( !CurrAttrib.Empty() )
183  this->Distance = CurrAttrib.AsReal();
184 
185  // Get the properties that need their own nodes
186  XML::Node NormalNode = SelfRoot.GetChild("Normal").GetFirstChild();
187  if( !NormalNode.Empty() )
188  this->Normal.ProtoDeSerialize(NormalNode);
189  }else{
190  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + Plane::GetSerializableName() + ": Not Version 1.");
191  }
192  }else{
193  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,Plane::GetSerializableName() + " was not found in the provided XML node, which was expected.");
194  }
195  }
