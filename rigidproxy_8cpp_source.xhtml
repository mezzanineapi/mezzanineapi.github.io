111  void RigidProxy::SetCollisionShape(CollisionShape* Shape)
112  {
113  if( Shape != NULL ) {
114  if(CollisionShape::ST_StaticTriMesh != Shape->GetType()) {
115  Real Mass = this->PhysicsRigidBody->getInvMass();
116  if(0 != Mass) {
117  Mass = 1 / Mass;
118  }
119  btVector3 Inertia(0,0,0);
120  Shape->_GetInternalShape()->calculateLocalInertia(Mass,Inertia);
121  this->PhysicsRigidBody->setMassProps(Mass,Inertia);
122  this->PhysicsProxy::SetCollisionShape(Shape);
123  this->PhysicsRigidBody->updateInertiaTensor();
124  }else{
125  this->PhysicsProxy::SetCollisionShape(Shape);
126  }
127  CollisionShapeManager::GetSingletonPtr()->StoreShape(Shape);
128  }else{
129  this->PhysicsProxy::SetCollisionShape(Shape);
130  }
131  }

239  void RigidProxy::ProtoSerializeProperties(XML::Node& SelfRoot) const
240  {
241  this->PhysicsProxy::ProtoSerialize(SelfRoot);
242  // We're at the base implementation, so no calling of child implementations
243  XML::Node PropertiesNode = SelfRoot.AppendChild( RigidProxy::GetSerializableName() + "Properties" );
244 
245  if( PropertiesNode.AppendAttribute("Version").SetValue("1") &&
246  PropertiesNode.AppendAttribute("Mass").SetValue( this->GetMass() ) &&
247  PropertiesNode.AppendAttribute("LinearDamping").SetValue( this->GetLinearDamping() ) &&
248  PropertiesNode.AppendAttribute("AngularDamping").SetValue( this->GetAngularDamping() ) &&
249  PropertiesNode.AppendAttribute("IsInWorld").SetValue( this->IsInWorld() ? "true" : "false" ) )
250  {
251  XML::Node LinVelNode = PropertiesNode.AppendChild("LinearVelocty");
252  this->GetLinearVelocity().ProtoSerialize( LinVelNode );
253 
254  XML::Node AngVelNode = PropertiesNode.AppendChild("AngularVelocity");
255  this->GetAngularVelocity().ProtoSerialize( AngVelNode );
256 
257  XML::Node LinFactNode = PropertiesNode.AppendChild("LinearFactor");
258  this->GetLinearMovementFactor().ProtoSerialize( LinFactNode );
259 
260  XML::Node AngFactNode = PropertiesNode.AppendChild("AngularFactor");
261  this->GetAngularMovementFactor().ProtoSerialize( AngFactNode );
262 
263  XML::Node ForceNode = PropertiesNode.AppendChild("Force");
264  this->GetAppliedForce().ProtoSerialize( ForceNode );
265 
266  XML::Node TorqueNode = PropertiesNode.AppendChild("Torque");
267  this->GetAppliedTorque().ProtoSerialize( TorqueNode );
268 
269  XML::Node GravityNode = PropertiesNode.AppendChild("Gravity");
270  this->GetGravity().ProtoSerialize( GravityNode );
271 
272  return;
273  }else{
274  SerializeError("Create XML Attribute Values",RigidProxy::GetSerializableName() + "Properties",true);
275  }

278  void RigidProxy::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
279  {
280  this->PhysicsRigidBody->clearForces();
281  this->PhysicsProxy::ProtoDeSerialize(SelfRoot);
282  // We're at the base implementation, so no calling of child implementations
283  XML::Attribute CurrAttrib;
284  XML::Node PropertiesNode = SelfRoot.GetChild( RigidProxy::GetSerializableName() + "Properties" );
285 
286  if( !PropertiesNode.Empty() ) {
287  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
288  Real LinDam = 0, AngDam = 0;
289  bool InWorld = false;
290 
291  CurrAttrib = PropertiesNode.GetAttribute("Mass");
292  if( !CurrAttrib.Empty() )
293  this->SetMass( CurrAttrib.AsReal() );
294 
295  CurrAttrib = PropertiesNode.GetAttribute("LinearDamping");
296  if( !CurrAttrib.Empty() )
297  LinDam = CurrAttrib.AsReal();
298 
299  CurrAttrib = PropertiesNode.GetAttribute("AngularDamping");
300  if( !CurrAttrib.Empty() )
301  AngDam = CurrAttrib.AsReal();
302 
303  CurrAttrib = PropertiesNode.GetAttribute("IsInWorld");
304  if( !CurrAttrib.Empty() )
305  InWorld = StringTools::ConvertToBool( CurrAttrib.AsString() );
306 
307  this->SetDamping(LinDam,AngDam);
308 
309  // Get the properties that need their own nodes
310  XML::Node LinVelNode = PropertiesNode.GetChild("LinearVelocty").GetFirstChild();
311  if( !LinVelNode.Empty() ) {
312  Vector3 LinVel(LinVelNode);
313  this->SetLinearVelocity(LinVel);
314  }
315 
316  XML::Node AngVelNode = PropertiesNode.GetChild("AngularVelocity").GetFirstChild();
317  if( !AngVelNode.Empty() ) {
318  Vector3 AngVel(AngVelNode);
319  this->SetAngularVelocity(AngVel);
320  }
321 
322  XML::Node LinFactNode = PropertiesNode.GetChild("LinearFactor").GetFirstChild();
323  if( !LinFactNode.Empty() ) {
324  Vector3 LinFact(LinFactNode);
325  this->SetLinearMovementFactor(LinFact);
326  }
327 
328  XML::Node AngFactNode = PropertiesNode.GetChild("AngularFactor").GetFirstChild();
329  if( !AngFactNode.Empty() ) {
330  Vector3 AngFact(AngFactNode);
331  this->SetAngularMovementFactor(AngFact);
332  }
333 
334  XML::Node ForceNode = PropertiesNode.GetChild("Force").GetFirstChild();
335  if( !ForceNode.Empty() ) {
336  Vector3 Force(ForceNode);
337  this->ApplyForce(Force);
338  }
339 
340  XML::Node TorqueNode = PropertiesNode.GetChild("Torque").GetFirstChild();
341  if( !TorqueNode.Empty() ) {
342  Vector3 Torque(TorqueNode);
343  this->ApplyTorque(Torque);
344  }
345 
346  XML::Node GravityNode = PropertiesNode.GetChild("Gravity").GetFirstChild();
347  if( !GravityNode.Empty() ) {
348  Vector3 Gravity(GravityNode);
349  this->SetGravity(Gravity);
350  }
351 
352  if( InWorld ) {
353  this->AddToWorld();
354  }
355  }else{
356  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (RigidProxy::GetSerializableName() + "Properties" ) + ": Not Version 1.");
357  }
358  }else{
359  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,RigidProxy::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
360  }
361  }
