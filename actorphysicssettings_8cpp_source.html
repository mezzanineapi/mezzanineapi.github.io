70  void ActorBasePhysicsSettings::SetCCDParams(const Real& MotionThreshold, const Real& SweptSphereRadius)
71  {
72  if(0==MotionThreshold)
73  {
74  WorldObjectCO->setCcdMotionThreshold(MotionThreshold);
75  WorldObjectCO->setCcdSweptSphereRadius(MotionThreshold);
76  }else{
77  WorldObjectCO->setCcdMotionThreshold(MotionThreshold);
78  if(0==SweptSphereRadius)
79  {
80  if(WorldObjectShape)
81  {
82  btTransform Trans;
83  btVector3 AABBmin, AABBmax, AABBsize;
84  WorldObjectShape->_GetInternalShape()->getAabb(Trans,AABBmin,AABBmax);
85  AABBsize = AABBmax - AABBmin;
86  Real ResultRadius = AABBsize.getX() < AABBsize.getY() ? (AABBsize.getX() < AABBsize.getZ() ? AABBsize.getX() : AABBsize.getZ()) : (AABBsize.getY() < AABBsize.getZ() ? AABBsize.getY() : AABBsize.getZ());
87  WorldObjectCO->setCcdSweptSphereRadius(ResultRadius*0.2);
88  }else{
89  WorldObjectCO->setCcdSweptSphereRadius(1);
90  }
91  }else{
92  WorldObjectCO->setCcdSweptSphereRadius(SweptSphereRadius);
93  }
94  }
95  }

215  void ActorRigidPhysicsSettings::ClearStickyContacts()
216  {
217  if(0 == StickyContacts->StickyConstraints.size())
218  return;
219  btDiscreteDynamicsWorld* BulletWorld = Entresol::GetSingletonPtr()->GetPhysicsManager()->_GetPhysicsWorldPointer();
220  for( std::vector<StickyConstraint*>::iterator SCit = StickyContacts->StickyConstraints.begin() ; SCit != StickyContacts->StickyConstraints.end() ; ++SCit )
221  {
222  BulletWorld->removeConstraint((*SCit)->GetConstraintBase());
223 
224  ActorRigid* OtherActor = (*SCit)->GetActorA() != Parent ? (*SCit)->GetActorA() : (*SCit)->GetActorB();
225  StickyData* OtherSettings = OtherActor->GetPhysicsSettings()->GetStickyData();
226  for( std::vector<StickyConstraint*>::iterator SCit2 = OtherSettings->StickyConstraints.begin() ; SCit2 != OtherSettings->StickyConstraints.end() ; ++SCit2 )
227  {
228  if( (*SCit) == (*SCit2) )
229  {
230  OtherSettings->StickyConstraints.erase(SCit2);
231  break;
232  }
233  }
234 
235  delete (*SCit);
236  }
237  StickyContacts->StickyConstraints.clear();
238  StickyContacts->CreationQueue.clear();
239  }
