487  void EnclosedRenderableContainerWidget::SetArea(const Vector2& Size)
488  {
489  Vector2 OldSize = WidgetBack->GetActualSize();
490  Vector2 Position = WidgetBack->GetActualPosition();
491  WidgetBack->SetActualSize(Size);
492  for( AreaRenderableInfoIterator it = AreaRenderables.begin() ; it != AreaRenderables.end() ; it++ )
493  {
494  (*it).Offset = CalculateOffset(Size,OldSize,(*it).Offset,(*it).Anchor);
495  if((*it).Tether != RT_TetherNone)
496  (*it).Object->SetActualSize(CalculateSize(Size,OldSize,(*it).Object->GetActualSize(),(*it).Tether));
497  (*it).Object->SetActualPosition(Position+(*it).Offset);
498  }
499  for( WidgetInfoIterator it = Widgets.begin() ; it != Widgets.end() ; it++ )
500  {
501  (*it).Offset = CalculateOffset(Size,OldSize,(*it).Offset,(*it).Anchor);
502  if((*it).Tether != RT_TetherNone)
503  (*it).Object->SetActualSize(CalculateSize(Size,OldSize,(*it).Object->GetActualSize(),(*it).Tether));
504  (*it).Object->SetActualPosition(Position+(*it).Offset);
505  }

508  Vector2 EnclosedRenderableContainerWidget::CalculateOffset(const Vector2& NewSize, const Vector2& OldSize, const Vector2& EleOffset, const UI::ResizeableAnchor& Anchor)
509  {
510  Vector2 NewOffset;
511  switch (Anchor)
512  {
513  case RA_AnchorTopLeft:
514  {
515  NewOffset = EleOffset;
516  break;
517  }
518  case RA_AnchorTop:
519  {
520  NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
521  NewOffset.Y = EleOffset.Y;
522  break;
523  }
524  case RA_AnchorTopRight:
525  {
526  NewOffset.X = EleOffset.X + (NewSize.X - OldSize.X);
527  NewOffset.Y = EleOffset.Y;
528  break;
529  }
530  case RA_AnchorLeft:
531  {
532  NewOffset.X = EleOffset.X;
533  NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
534  break;
535  }
536  case RA_AnchorMiddle:
537  {
538  NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
539  NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
540  break;
541  }
542  case RA_AnchorRight:
543  {
544  NewOffset.X = EleOffset.X + (NewSize.X - OldSize.X);
545  NewOffset.Y = EleOffset.Y + ((NewSize.Y - OldSize.Y) * 0.5);
546  break;
547  }
548  case RA_AnchorBottomLeft:
549  {
550  NewOffset.X = EleOffset.X;
551  NewOffset.Y = EleOffset.Y + (NewSize.Y - OldSize.Y);
552  break;
553  }
554  case RA_AnchorBottom:
555  {
556  NewOffset.X = EleOffset.X + ((NewSize.X - OldSize.X) * 0.5);
557  NewOffset.Y = EleOffset.Y + (NewSize.Y - OldSize.Y);
558  break;
559  }
560  case RA_AnchorBottomRight:
561  {
562  NewOffset = EleOffset + (NewSize - OldSize);
563  break;
564  }
565  default:
566  return NewOffset;
567  }
568  return NewOffset;
569  }

571  Vector2 EnclosedRenderableContainerWidget::CalculateSize(const Vector2& NewSize, const Vector2& OldSize, const Vector2& EleSize, const UI::ResizeableTether& Tether)
572  {
573  Vector2 NewEleSize;
574  switch (Tether)
575  {
576  case RT_TetherBoth:
577  {
578  NewEleSize = EleSize + (NewSize - OldSize);
579  break;
580  }
581  case RT_TetherNone:
582  {
583  NewEleSize = EleSize;
584  break;
585  }
586  case RT_TetherHorizontal:
587  {
588  NewEleSize.X = EleSize.X + (NewSize.X - OldSize.X);
589  NewEleSize.Y = EleSize.Y;
590  break;
591  }
592  case RT_TetherVertical:
593  {
594  NewEleSize.X = EleSize.X;
595  NewEleSize.Y = EleSize.Y + (NewSize.Y - OldSize.Y);
596  break;
597  }
598  default:
599  return NewEleSize;
600  }
601  return NewEleSize;
602  }
