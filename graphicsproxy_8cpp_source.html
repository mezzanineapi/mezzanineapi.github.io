117  void GraphicsProxy::ProtoSerializeProperties(XML::Node& SelfRoot) const
118  {
119  // We're at the base implementation, so no calling of child implementations
120  XML::Node PropertiesNode = SelfRoot.AppendChild( GraphicsProxy::GetSerializableName() + "Properties" );
121 
122  if( PropertiesNode.AppendAttribute("Version").SetValue("1") &&
123  PropertiesNode.AppendAttribute("Visible").SetValue( this->GetVisible() ? "true" : "false" ) &&
124  PropertiesNode.AppendAttribute("CastShadows").SetValue( this->GetCastShadows() ? "true" : "false" ) &&
125  PropertiesNode.AppendAttribute("LightMask").SetValue( this->GetLightMask() ) &&
126  PropertiesNode.AppendAttribute("RenderDistance").SetValue( this->GetRenderDistance() ) )
127  {
128  XML::Node LocationNode = PropertiesNode.AppendChild("Location");
129  this->GetLocation().ProtoSerialize( LocationNode );
130  XML::Node OrientationNode = PropertiesNode.AppendChild("Orientation");
131  this->GetOrientation().ProtoSerialize( OrientationNode );
132  XML::Node ScaleNode = PropertiesNode.AppendChild("Scale");
133  this->GetScale().ProtoSerialize( ScaleNode );
134 
135  return;
136  }else{
137  SerializeError("Create XML Attribute Values",GraphicsProxy::GetSerializableName() + "Properties",true);
138  }
139  }

146  void GraphicsProxy::ProtoDeSerializeProperties(const XML::Node& SelfRoot)
147  {
148  // We're at the base implementation, so no calling of child implementations
149  XML::Attribute CurrAttrib;
150  XML::Node PropertiesNode = SelfRoot.GetChild( GraphicsProxy::GetSerializableName() + "Properties" );
151 
152  if( !PropertiesNode.Empty() ) {
153  if(PropertiesNode.GetAttribute("Version").AsInt() == 1) {
154  CurrAttrib = PropertiesNode.GetAttribute("Visible");
155  if( !CurrAttrib.Empty() )
156  this->SetVisible( StringTools::ConvertToBool( CurrAttrib.AsString() ) );
157 
158  CurrAttrib = PropertiesNode.GetAttribute("CastShadows");
159  if( !CurrAttrib.Empty() )
160  this->SetCastShadows( StringTools::ConvertToBool( CurrAttrib.AsString() ) );
161 
162  CurrAttrib = PropertiesNode.GetAttribute("LightMask");
163  if( !CurrAttrib.Empty() )
164  this->SetLightMask( CurrAttrib.AsWhole() );
165 
166  CurrAttrib = PropertiesNode.GetAttribute("RenderDistance");
167  if( !CurrAttrib.Empty() )
168  this->SetRenderDistance( CurrAttrib.AsReal() );
169 
170  // Get the properties that need their own nodes
171  XML::Node PositionNode = PropertiesNode.GetChild("Location").GetFirstChild();
172  if( !PositionNode.Empty() ) {
173  Vector3 Loc(PositionNode);
174  this->SetLocation(Loc);
175  }
176 
177  XML::Node OrientationNode = PropertiesNode.GetChild("Orientation").GetFirstChild();
178  if( !PositionNode.Empty() ) {
179  Quaternion Rot(OrientationNode);
180  this->SetOrientation(Rot);
181  }
182 
183  XML::Node ScaleNode = PropertiesNode.GetChild("Scale").GetFirstChild();
184  if( !PositionNode.Empty() ) {
185  Vector3 Scale(ScaleNode);
186  this->SetScale(Scale);
187  }
188  }else{
189  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (GraphicsProxy::GetSerializableName() + "Properties" ) + ": Not Version 1.");
190  }
191  }else{
192  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,GraphicsProxy::GetSerializableName() + "Properties" + " was not found in the provided XML node, which was expected.");
193  }
194  }
