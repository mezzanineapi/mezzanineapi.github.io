53  /// @class EntityInternalData
54  /// @brief This class stores the internal data of an Entity.
55  ///////////////////////////////////////
56  struct EntityInternalData
57  {
58  /// @internal
59  /// @brief Pointer to the ogre Entity from which this class gets it's functionality.
60  Ogre::Entity* OgreEntity;
61 
62  /// @internal
63  /// @brief Pointer to the ogre Scenenode to which this object is attached.
64  Ogre::SceneNode* OgreNode;
65 
66  /// @internal
67  /// @brief Pointer to the manager that created this class.
68  SceneManager* Manager;
69 
70  /// @internal
71  /// @brief Creates all the necessary data structures for the Entity.
72  EntityInternalData(const String& Name, const String& MeshName, const String& Group, SceneManager* manager)
73  {
74  this->Manager = manager;
75  Ogre::SceneManager* OgreManager = Manager->GetGraphicsWorldPointer();
76  OgreEntity = OgreManager->createEntity(Name,MeshName,Group);
77  OgreNode = OgreManager->createSceneNode();
78  OgreManager->getRootSceneNode()->addChild(OgreNode);
79  }
80 
81  /// @internal
82  /// @brief Cleans up after the Entity.
83  ~EntityInternalData()
84  {
85  Manager->GetGraphicsWorldPointer()->destroyEntity(OgreEntity);
86  Manager->GetGraphicsWorldPointer()->destroySceneNode(OgreNode);
87  }
88  };//EntityInternalData
89  }//internal
