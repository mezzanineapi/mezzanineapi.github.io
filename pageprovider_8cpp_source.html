82  void PageProvider::ProtoSerializePageData(XML::Node& SelfRoot) const
83  {
84  if( this->Container != NULL ) {
85  XML::Node PageDataNode = SelfRoot.AppendChild( PageProvider::GetSerializableName() + "PageData" );
86 
87  if( PageDataNode.AppendAttribute("Version").SetValue("1") &&
88  PageDataNode.AppendAttribute("ContainerName").SetValue( this->Container->GetName() ) &&
89  PageDataNode.AppendAttribute("ProviderConfig").SetValue( static_cast<Whole>( this->Container->GetProviderConfig(this) ) ) )
90  {
91  return;
92  }else{
93  SerializeError("Create XML Attribute Values",PageProvider::GetSerializableName() + "PageData",true);
94  }
95  }
96  }

98  void PageProvider::ProtoDeSerializePageData(const XML::Node& SelfRoot)
99  {
100  XML::Attribute CurrAttrib;
101  XML::Node PageDataNode = SelfRoot.GetChild( PageProvider::GetSerializableName() + "PageData" );
102 
103  if( !PageDataNode.Empty() ) {
104  if(PageDataNode.GetAttribute("Version").AsInt() == 1) {
105  String ContainerName;
106  PagedContainer::ProviderMode ProviderConfig;
107 
108  CurrAttrib = PageDataNode.GetAttribute("ContainerName");
109  if( !CurrAttrib.Empty() )
110  ContainerName = CurrAttrib.AsString();
111 
112  CurrAttrib = PageDataNode.GetAttribute("ProviderConfig");
113  if( !CurrAttrib.Empty() )
114  ProviderConfig = static_cast<PagedContainer::ProviderMode>( CurrAttrib.AsWhole() );
115 
116  if( !ContainerName.empty() ) {
117  PagedContainer* ToSet = static_cast<PagedContainer*>( this->ParentScreen->GetWidget(ContainerName) );
118  if( ToSet != NULL ) {
119  switch( ProviderConfig )
120  {
121  case PagedContainer::PM_Single_X: ToSet->SetXProvider(this); break;
122  case PagedContainer::PM_Single_Y: ToSet->SetYProvider(this); break;
123  case PagedContainer::PM_Single_XY: ToSet->SetProviders(this,this); break;
124  default: /* Do Nothing */ break;
125  }
126  }
127  }
128  }else{
129  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + (PageProvider::GetSerializableName() + "PageData") + ": Not Version 1.");
130  }
131  }else{
132  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,PageProvider::GetSerializableName() + "PageData" + " was not found in the provided XML node, which was expected.");
133  }
134  }
