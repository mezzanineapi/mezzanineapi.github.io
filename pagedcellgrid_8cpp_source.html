74  void PagedCellGrid::CalculateCellsPerPage()
75  {
76  Real XSpace = RelSize.X - (EdgeSpacing.X * 2);
77  Real YSpace = RelSize.Y - (EdgeSpacing.Y * 2);
78  XCellsPerPage = 1;
79  YCellsPerPage = 1;
80 
81  Real XCursor = FixedCellSize.X;
82  Real YCursor = FixedCellSize.Y;
83  bool DoNotBreak = true;
84  while(DoNotBreak)
85  {
86  if(XCursor + (FixedCellSize.X + CellSpacing.X) < XSpace)
87  {
88  XCursor+=(FixedCellSize.X + CellSpacing.X);
89  XCellsPerPage++;
90  }
91  else if(XCursor + FixedCellSize.X < XSpace)
92  {
93  XCursor+=FixedCellSize.X;
94  XCellsPerPage++;
95  }
96  else
97  {
98  DoNotBreak = false;
99  }
100  }
101  DoNotBreak = true;
102  while(DoNotBreak)
103  {
104  if(YCursor + (FixedCellSize.Y + CellSpacing.Y) < YSpace)
105  {
106  YCursor+=(FixedCellSize.Y + CellSpacing.Y);
107  YCellsPerPage++;
108  }
109  else if(YCursor + FixedCellSize.Y < YSpace)
110  {
111  YCursor+=FixedCellSize.Y;
112  YCellsPerPage++;
113  }
114  else
115  {
116  DoNotBreak = false;
117  }
118  }
119  }

121  std::pair<Whole,Whole> PagedCellGrid::FindTopLeftOfPage(const Whole& Page)
122  {
123  Whole TotalRows = TheGrid.size();
124  Whole TotalColumns = TheGrid.at(0)->size();
125 
126  Whole XPages = TotalColumns / XCellsPerPage;
127  if(TotalColumns % XCellsPerPage)
128  XPages++;
129  Whole YPages = TotalRows / YCellsPerPage;
130  if(TotalRows % YCellsPerPage)
131  YPages++;
132 
133  MaxPages = XPages * YPages;
134 
135  Whole RRow = 0;
136  Whole RColumn = 0;
137  Whole page = Page;
138  bool DoNotBreak = true;
139  while(DoNotBreak)
140  {
141  if(page > XPages)
142  {
143  page-=XPages;
144  RRow+=YCellsPerPage;
145  }else{
146  //if(1==Page)
147  // RColumn = 0;
148  //else
149  RColumn+=((XCellsPerPage * page) - XCellsPerPage);
150  DoNotBreak = false;
151  }
152  }
153  return std::pair<Whole,Whole>(RRow,RColumn);
154  }

162  void PagedCellGrid::DrawGrid(const Vector2& WinDim)
163  {
164  if(Cells.empty())
165  return;
166  RegenerateGrid();
167  for( Whole X = 0 ; X < VisibleCells.size() ; X++ )
168  VisibleCells[X]->Hide();
169  VisibleCells.clear();
170 
171  std::pair<Whole,Whole> Result = FindTopLeftOfPage(CurrentPage);
172  PageSpinner->SetValueLimits(1,MaxPages);
173  Whole CurrentRow = Result.first;
174  Whole CurrentColumn = Result.second;
175  const Whole StartRow = CurrentRow;
176  const Whole StartColumn = CurrentColumn;
177 
178  Vector2 ActPos = GetActualPosition();
179  Vector2 ActSize = GetActualSize();
180  Whole CursorXPos = (Whole)(ActPos.X + EdgeSpacing.X);
181  Whole CursorYPos = (Whole)(ActPos.Y + EdgeSpacing.Y);
182  const Whole StartXPos = CursorXPos;
183  const Whole StartYPos = CursorYPos;
184 
185  UI::Cell* CurrentCell = NULL;
186  bool StillHasRoom = true;
187 
188  switch (Ordering)
189  {
190  case CG_Horizontal_Vertical_Ascending:
191  case CG_Horizontal_Vertical_Decending:
192  {
193  if(ActSize.Y < (FixedCellSize.Y + (EdgeSpacing.Y * 2)))
194  return;
195  while(StillHasRoom)
196  {
197  CurrentCell = GetCell(CurrentRow,CurrentColumn);
198  if(!CurrentCell)
199  {
200  CursorYPos+=FixedCellSize.Y + CellSpacing.Y;
201  CurrentRow++;
202  CursorXPos = StartXPos;
203  CurrentColumn = StartColumn;
204  if(CurrentRow >= GetMaxRows())
205  StillHasRoom = false;
206  continue;
207  }
208  Vector2 CellSize = CurrentCell->GetActualSize();
209  if((Real)CursorXPos + CellSize.X > (ActPos.X + ActSize.X) - EdgeSpacing.X)
210  {
211  CursorYPos+=FixedCellSize.Y + CellSpacing.Y;
212  CurrentRow++;
213  CursorXPos = StartXPos;
214  CurrentColumn = StartColumn;
215  if(CursorYPos+FixedCellSize.Y > (ActPos.Y + ActSize.Y) - EdgeSpacing.Y)
216  StillHasRoom = false;
217  continue;
218  }
219  CurrentCell->SetActualPosition(Vector2((Real)CursorXPos,(Real)CursorYPos));
220  if(Visible)
221  CurrentCell->Show();
222  VisibleCells.push_back(CurrentCell);
223  CursorXPos+=(Whole)(CellSize.X + CellSpacing.X);
224  CurrentColumn++;
225  }
226  break;
227  }
228  case CG_Vertical_Horizontal_Ascending:
229  case CG_Vertical_Horizontal_Decending:
230  {
231  if(ActSize.X < (FixedCellSize.X + (EdgeSpacing.X * 2)))
232  return;
233  while(StillHasRoom)
234  {
235  CurrentCell = GetCell(CurrentRow,CurrentColumn);
236  if(!CurrentCell)
237  {
238  CursorXPos+=FixedCellSize.X + CellSpacing.X;
239  CurrentColumn++;
240  CursorYPos = StartYPos;
241  CurrentRow = StartRow;
242  if(CurrentColumn >= GetMaxColumns())
243  StillHasRoom = false;
244  continue;
245  }
246  Vector2 CellSize = CurrentCell->GetActualSize();
247  if((Real)CursorYPos + CellSize.Y > (ActPos.Y + ActSize.Y) - EdgeSpacing.Y)
248  {
249  CursorXPos+=FixedCellSize.X + CellSpacing.X;
250  CurrentColumn++;
251  CursorYPos = StartYPos;
252  CurrentRow = StartRow;
253  if(CursorXPos+FixedCellSize.X > (ActPos.X + ActSize.X) - EdgeSpacing.X)
254  StillHasRoom = false;
255  continue;
256  }
257  CurrentCell->SetActualPosition(Vector2((Real)CursorXPos,(Real)CursorYPos));
258  if(Visible)
259  CurrentCell->Show();
260  VisibleCells.push_back(CurrentCell);
261  CursorYPos+=(Whole)(CellSize.Y + CellSpacing.Y);
262  CurrentRow++;
263  }
264  break;
265  }
266  }
267  }
