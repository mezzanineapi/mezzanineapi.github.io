129  {
130  XML::Node SelfRoot = ParentNode.AppendChild( Ray::GetSerializableName() );
131 
132  if( SelfRoot.AppendAttribute("Version").SetValue("1") )
133  {
134  XML::Node OriginNode = SelfRoot.AppendChild("Origin");
135  this->Origin.ProtoSerialize( OriginNode );

146  void Ray::ProtoDeSerialize(const XML::Node& SelfRoot)
147  {
148  if( String(SelfRoot.Name()) == Ray::GetSerializableName() ) {
149  if(SelfRoot.GetAttribute("Version").AsInt() == 1) {
150  // Get the properties that need their own nodes
151  XML::Node OriginNode = SelfRoot.GetChild("Origin").GetFirstChild();
152  if( !OriginNode.Empty() )
153  this->Origin.ProtoDeSerialize(OriginNode);
154 
155  XML::Node DestinationNode = SelfRoot.GetChild("Destination").GetFirstChild();
156  if( !DestinationNode.Empty() )
157  this->Destination.ProtoDeSerialize(DestinationNode);
158  }else{
159  MEZZ_EXCEPTION(Exception::INVALID_VERSION_EXCEPTION,"Incompatible XML Version for " + Ray::GetSerializableName() + ": Not Version 1.");
160  }
161  }else{
162  MEZZ_EXCEPTION(Exception::II_IDENTITY_NOT_FOUND_EXCEPTION,Ray::GetSerializableName() + " was not found in the provided XML node, which was expected.");
163  }
