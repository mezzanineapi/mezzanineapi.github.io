197  void MarkupText::_GenerateWords()
198  {
199  _ClearWords();
200  Word* CurrWord = NULL;
201  bool MarkupMode = false;
202  bool FixedWidth = false;
203  Glyph* CurrGlyph = NULL;
204  GlyphData* CurrGlyphData = DefaultGlyphSet;
205  ColourValue Colour = ParentScreen->GetMarkupColour(0,PriAtlas);
206  Real LineHeight = CurrGlyphData->LineHeight;
207  Real Kerning = 0;
208  unsigned int ThisChar = 0, LastChar = 0;
209 
210  for( Whole Index = 0 ; Index < Text.length() ; Index++ )
211  {
212  ThisChar = Text[Index];
213 
214  if( ThisChar == ' ' )
215  {
216  LastChar = ThisChar;
217  if(CurrWord)
218  {
219  CurrWord->GlyphDataSpacing = CurrGlyphData->SpaceLength;
220  Words.push_back(CurrWord);
221  CurrWord = NULL;
222  }
223  continue;
224  }
225 
226  if( ThisChar == '\n' )
227  {
228  LastChar = ThisChar;
229  if(CurrWord)
230  {
231  //CurrWord->NewlineAfter = true;
232  Words.push_back(CurrWord);
233  CurrWord = NULL;
234  }
235  CurrWord = new Word();
236  CurrWord->NewlineBefore = true;
237  continue;
238  }
239 
240  if( ThisChar < CurrGlyphData->RangeBegin || ThisChar > CurrGlyphData->RangeEnd )
241  {
242  LastChar = 0;
243  continue;
244  }
245 
246  if( ThisChar == '%' && !MarkupMode )
247  {
248  MarkupMode = true;
249  continue;
250  }
251 
252  if( MarkupMode == true )
253  {
254  if (ThisChar == '%')
255  {
256  // Escape Character.
257  }else{
258  MarkupMode = false;
259 
260  // Check if the user wants a different colour to use for all further glyphs.
261  if(ThisChar >= '0' && ThisChar <= '9')
262  {
263  Colour = ParentScreen->GetMarkupColour(ThisChar - 48,PriAtlas);
264  }
265  // Check if the user wants to revert the current selected colour back to default.
266  else if(ThisChar == 'R' || ThisChar == 'r')
267  {
268  Colour = ParentScreen->GetMarkupColour(0,PriAtlas);
269  }
270  // Fixed Width? Like Notepad?
271  else if(ThisChar == 'M' || ThisChar == 'm')
272  {
273  FixedWidth = !FixedWidth;
274  }
275  // Check if the user wants to change the glyph set being rendered.
276  else if( ThisChar == '@' )
277  {
278  MarkupMode = false;
279  bool FoundIt = false;
280  Whole Begin = Index;
281  while( Index < Text.size() )
282  {
283  if( Text[Index] == '%' )
284  {
285  FoundIt = true;
286  break;
287  }
288  Index++;
289  }
290 
291  if( FoundIt == false )
292  return;
293 
294  UInt32 ID = StringTools::ConvertToUInt32(Text.substr(Begin+1, Index - Begin - 1));
295  CurrGlyphData = ParentScreen->GetGlyphData(ID,PriAtlas);
296  if( CurrGlyphData == 0 )
297  return;
298  continue;
299  }
300  // Check to see if the user wants to change the glyph with a sprite.
301  else if( ThisChar == ':' )
302  {
303  MarkupMode = false;
304  bool FoundIt = false;
305  size_t Begin = Index;
306  size_t Middle = 0;
307  while( Index < Text.size() )
308  {
309  if( Text[Index] == ',' )
310  {
311  Middle = Index;
312  }
313  if( Text[Index] == '%' )
314  {
315  FoundIt = true;
316  break;
317  }
318  Index++;
319  }
320 
321  if( FoundIt == false )
322  return;
323 
324  String SpriteName = Text.substr(Begin+1, Middle - Begin - 1);
325  String SpriteAtlas = Text.substr(Middle+1, Index - Middle - 1);
326  Sprite* NewSprite = ParentScreen->GetSprite(SpriteName,SpriteAtlas);
327  if( NewSprite == 0 )
328  continue;
329 
330  if( !CurrWord )
331  CurrWord = new Word();
332 
333  Character* c = new Character();
334  c->Index = Index;
335  c->Colour = Colour;
336  c->CharSprite = NewSprite;
337  c->Atlas = NewSprite->Atlas;
338 
339  CurrWord->AddCharacter(c,CharScaling);
340 
341  continue;
342  }
343  continue;
344  }
345  MarkupMode = false;
346  }
347 
348  CurrGlyph = CurrGlyphData->GetGlyph(ThisChar);
349 
350  if( !CurrWord )
351  CurrWord = new Word();
352 
353  if( !FixedWidth )
354  {
355  Kerning = CurrGlyph->GetKerning(LastChar);
356  if (Kerning == 0)
357  Kerning = CurrGlyphData->LetterSpacing;
358  }
359 
360  Character* c = new Character();
361  c->Index = Index;
362  c->Colour = Colour;
363  c->CharGlyph = CurrGlyph;
364  c->Kerning = Kerning;
365  c->Atlas = CurrGlyphData->Atlas;
366 
367  CurrWord->AddCharacter(c,CharScaling);
368 
369  LastChar = ThisChar;
370  }
371 
372  if( CurrWord )
373  {
374  if(CurrWord->Characters.size()) Words.push_back(CurrWord);
375  else delete CurrWord;
376  }
377  }

379  void MarkupText::_PlaceWords()
380  {
381  if(Words.empty())
382  return;
383 
384  Real CursorX = ActPosition.X, CursorY = 0;
385  Real TexelX = ParentScreen->GetTexelOffsetX(), TexelY = ParentScreen->GetTexelOffsetY();
386  Real Kerning = 0, Top = 0, Left = 0, Bottom = 0, Right = 0;
387 
388  std::vector<Real> Lineheights;
389  Real CurrLineHeight = 0;
390  for( UInt32 words = 0 ; words < Words.size() ; ++words )
391  {
392  Word* CurrWord = Words.at(words);
393  if(CursorX+CurrWord->Pxlength > ActPosition.X+ActSize.X)
394  {
395  CurrWord->NewlineBefore = true;
396  }
397  if(CurrWord->NewlineBefore)
398  {
399  Lineheights.push_back(CurrLineHeight);
400  CurrLineHeight = 0;
401  CursorX = ActPosition.X;
402  }else{
403  CursorX += (CurrWord->Pxlength + CurrWord->GlyphDataSpacing);
404  }
405  if(CurrWord->Pxheight > CurrLineHeight)
406  CurrLineHeight = CurrWord->Pxheight;
407  }
408  Lineheights.push_back(CurrLineHeight);
409 
410  CursorX = ActPosition.X;
411  CursorY = ActPosition.Y + Lineheights[0];
412  for( UInt32 words = 0, lineindex = 0 ; words < Words.size() ; ++words )
413  {
414  Word* CurrWord = Words[words];
415  UInt32 NumChars = CurrWord->Characters.size();
416 
417  if(CurrWord->NewlineBefore)
418  {
419  lineindex++;
420  CursorX = ActPosition.X;
421  CursorY += Lineheights[lineindex];
422  }// */
423 
424  for( UInt32 chars = 0 ; chars < NumChars ; ++chars )
425  {
426  Character* CurrChar = CurrWord->Characters[chars];
427 
428  if(CurrChar->CharGlyph)
429  {
430  Left = CursorX;
431  Top = CursorY - (CurrChar->CharGlyph->GlyphSize.Y * CharScaling) - TexelY;
432  Right = CursorX + (CurrChar->CharGlyph->GlyphSize.X * CharScaling) + TexelX;
433  Bottom = CursorY - (CurrChar->CharGlyph->VerticalOffset * CharScaling);
434  }
435  else if(CurrChar->CharSprite)
436  {
437  Left = CursorX - TexelX;
438  Top = CursorY - (CurrChar->CharSprite->SpriteSize.Y * CharScaling) - TexelY;
439  Right = CursorX + (CurrChar->CharSprite->SpriteSize.X * CharScaling) + TexelX;
440  Bottom = CursorY + TexelY;
441  }
442  else
443  {
444  // If this happens, just skip, cause I don't know wtf is going on.
445  continue;
446  }
447 
448  CurrChar->Positions[UI::QC_TopLeft].X = Left;
449  CurrChar->Positions[UI::QC_TopLeft].Y = Top;
450  CurrChar->Positions[UI::QC_TopRight].X = Right;
451  CurrChar->Positions[UI::QC_TopRight].Y = Top;
452  CurrChar->Positions[UI::QC_BottomLeft].X = Left;
453  CurrChar->Positions[UI::QC_BottomLeft].Y = Bottom;
454  CurrChar->Positions[UI::QC_BottomRight].X = Right;
455  CurrChar->Positions[UI::QC_BottomRight].Y = Bottom;
456  CurrChar->UVs[0] = (CurrChar->CharGlyph ? CurrChar->CharGlyph->AtlasCoords[0] : CurrChar->CharSprite->AtlasCoords[0]);
457  CurrChar->UVs[1] = (CurrChar->CharGlyph ? CurrChar->CharGlyph->AtlasCoords[1] : CurrChar->CharSprite->AtlasCoords[1]);
458  CurrChar->UVs[2] = (CurrChar->CharGlyph ? CurrChar->CharGlyph->AtlasCoords[2] : CurrChar->CharSprite->AtlasCoords[2]);
459  CurrChar->UVs[3] = (CurrChar->CharGlyph ? CurrChar->CharGlyph->AtlasCoords[3] : CurrChar->CharSprite->AtlasCoords[3]);
460 
461  if(CurrChar->CharGlyph) CursorX += ((CurrChar->CharGlyph->GlyphAdvance + CurrChar->Kerning) * CharScaling);
462  else if(CurrChar->CharSprite) CursorX += (CurrChar->CharSprite->SpriteSize.X * CharScaling);
463 
464  Characters.push_back(CurrChar);
465  }
466 
467  CursorX += (CurrWord->GlyphDataSpacing * CharScaling);
468
