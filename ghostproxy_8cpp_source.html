115  PhysicsProxy* GhostProxy::GetShapeOverlappingProxy(const UInt32 Index)
116  {
117  btCollisionWorld* PhysWorld = this->Manager->_GetPhysicsWorldPointer();
118 
119  btBroadphasePairArray& PairArray = this->PhysicsGhostBody->getOverlappingPairCache()->getOverlappingPairArray();
120  const btBroadphasePair& CachePair = PairArray[Index];
121  btBroadphasePair* WorldPair = PhysWorld->getPairCache()->findPair(CachePair.m_pProxy0,CachePair.m_pProxy1);
122 
123  if( WorldPair != NULL )
124  {
125  btManifoldArray ManifoldArray;
126  ManifoldArray.clear();
127 
128  if( WorldPair->m_algorithm )
129  WorldPair->m_algorithm->getAllContactManifolds(ManifoldArray);
130 
131  for( int ManifoldIndex = 0 ; ManifoldIndex < ManifoldArray.size() ; ++ManifoldIndex )
132  {
133  btPersistentManifold* Manifold = ManifoldArray[ManifoldIndex];
134  for( int ContactIndex = 0 ; ContactIndex < Manifold->getNumContacts() ; ++ContactIndex )
135  {
136  //const btManifoldPoint& pt = Manifold->getContactPoint(ContactIndex);
137  //if( pt.m_distance1 > 0 )
138  // return NULL;
139 
140  btCollisionObject* ColObj = ( Manifold->getBody0() != this->PhysicsGhostBody ? (btCollisionObject*)(Manifold->getBody0()) : (btCollisionObject*)(Manifold->getBody1()) );
141  return static_cast<PhysicsProxy*>( ColObj->getUserPointer() );
142  }
143  }
144  }else{
145  return NULL;
146  }
147  }
