55  Button::Button(ConstString& name, const RenderableRect& Rect, Screen* PScreen)
56  : Activatable(name,PScreen)
57  {
58  Type = Widget::W_Button;
59  Real LineHeight = 0.0;
60  if(Rect.Relative)
61  {
62  RelPosition = Rect.Position;
63  RelSize = Rect.Size;
64  LineHeight = Rect.Size.Y * 0.8;
65  }else{
66  RelPosition = Rect.Position / ParentScreen->GetViewportDimensions();
67  RelSize = Rect.Size / ParentScreen->GetViewportDimensions();
68  LineHeight = (Rect.Size.Y / ParentScreen->GetViewportDimensions().Y) * 0.8;
69  }
70  Clickable = ParentScreen->CreateCaption(Name+"Cl",Rect,LineHeight,"");
71  AddSubRenderable(0,Clickable);
72  }

139  {
140  WidgetResult Result = ViewportUpdateTool::UpdateWidget(this);
