390  Quaternion Vector3::GetRotationToAxis(const Vector3& Axis, const Vector3& FallBackAxis) const
391  {
392  Quaternion Ret;
393  Vector3 Vec1 = *this;
394  Vector3 Vec2 = Axis;
395  Vec1.Normalize();
396  Vec2.Normalize();
397 
398  Real Dot = Vec1.DotProduct(Vec2);
399  if( Dot >= 1.0 )
400  {
401  return Ret;
402  }
403  if( Dot < (1e-6 - 1.0) )
404  {
405  if( FallBackAxis != Vector3() )
406  {
407  Ret.SetFromAxisAngle(MathTools::Pi,FallBackAxis);
408  }else{
409  Vector3 CrossAxis = Vector3::Unit_X().CrossProduct(*this);
410  if(CrossAxis.IsZeroLength())
411  CrossAxis = Vector3::Unit_Y().CrossProduct(*this);
412  CrossAxis.Normalize();
413  Ret.SetFromAxisAngle(MathTools::Pi,CrossAxis);
414  }
415  }else{
416  Real Sqr = MathTools::Sqrt( (1+Dot)*2 );
417  Real InvSqr = 1 / Sqr;
418 
419  Vector3 Cross = Vec1.CrossProduct(Vec2);
420 
421  Ret.X = Cross.X * InvSqr;
422  Ret.Y = Cross.Y * InvSqr;
423  Ret.Z = Cross.Z * InvSqr;
424  Ret.W = Sqr * 0.5f;
425  Ret.Normalize();
426  }
427  return Ret;
428  }

504  Mezzanine::XML::Attribute VersionAttr = VecNode.AppendAttribute("Version");

507  Mezzanine::XML::Attribute ZAttr = VecNode.AppendAttribute("Z");
