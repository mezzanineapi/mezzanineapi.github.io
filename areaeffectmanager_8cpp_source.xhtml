217  void AreaEffectManager::Initialize()
218  {
219  if( !this->Initialized )
220  {
221  //WorldManager::Initialize();
222 
223  this->TheEntresol->GetScheduler().AddWorkUnitMain( this->AreaEffectUpdateWork, "AreaEffectUpdateWork" );
224 
225  Physics::PhysicsManager* PhysicsMan = this->TheEntresol->GetPhysicsManager();
226  if( PhysicsMan ) {
227  this->AreaEffectUpdateWork->AddDependency( PhysicsMan->GetSimulationWork() );
228  this->AreaEffectUpdateWork->AddDependency( PhysicsMan->GetWorldTriggerUpdateWork() );
229  this->AreaEffectUpdateWork->AddDependency( PhysicsMan->GetDebugDrawWork() );
230  }
231 
232  Mezzanine::ActorManager* ActorMan = this->TheEntresol->GetActorManager();
233  if( ActorMan ) {
234  this->AreaEffectUpdateWork->AddDependency( ActorMan->GetActorUpdateWork() );
235  }
236 
237  this->Initialized = true;
238  }
239  }
