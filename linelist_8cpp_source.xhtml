109  void LineList::_Redraw()
110  {
111  if(Dirty == false)
112  return;
113  RenderVertices.clear();
114  if(!Visible)
115  {
116  Dirty = false;
117  return;
118  }
119  if(Positions.size() < 2)
120  return;
121 
122  VertexData Temp;
123  Real HalfThickness = Thickness * 0.5;
124  Vector2 PerpNorm, LastLeft, LastRight, ThisLeft, ThisRight, UV = ParentScreen->GetSolidUV(PriAtlas);
125  size_t Index = 1;
126  for( ; Index < Positions.size() ; Index++ )
127  {
128  PerpNorm = (Positions[Index] - Positions[Index - 1]).Perpendicular().Normalize();
129  LastLeft = Positions[Index -1 ] - PerpNorm * HalfThickness;
130  LastRight = Positions[Index -1 ] + PerpNorm * HalfThickness;
131  ThisLeft = Positions[Index] - PerpNorm * HalfThickness;
132  ThisRight = Positions[Index] + PerpNorm * HalfThickness;
133 
134  // Triangle A
135  PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas); // Left/Bottom
136  PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas); // Right/Top
137  PushVertex(RenderVertices,Temp,LastLeft.X,LastLeft.Y,UV,Colour,PriAtlas); // Left/Top
138  // Triangle B
139  PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas); // Left/Bottom
140  PushVertex(RenderVertices,Temp,ThisRight.X,ThisRight.Y,UV,Colour,PriAtlas); // Right/Bottom
141  PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas); // Right/Top
142  }
143 
144  if(IsClosed)
145  {
146  Index = Positions.size() - 1;
147  PerpNorm = (Positions[0] - Positions[Index]).Perpendicular().Normalize();
148  LastLeft = Positions[Index] - PerpNorm * HalfThickness;
149  LastRight = Positions[Index] + PerpNorm * HalfThickness;
150  ThisLeft = Positions[0] - PerpNorm * HalfThickness;
151  ThisRight = Positions[0] + PerpNorm * HalfThickness;
152 
153  // Triangle A
154  PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas); // Left/Bottom
155  PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas); // Right/Top
156  PushVertex(RenderVertices,Temp,LastLeft.X,LastLeft.Y,UV,Colour,PriAtlas); // Left/Top
157  // Triangle B
158  PushVertex(RenderVertices,Temp,LastRight.X,LastRight.Y,UV,Colour,PriAtlas); // Left/Bottom
159  PushVertex(RenderVertices,Temp,ThisRight.X,ThisRight.Y,UV,Colour,PriAtlas); // Right/Bottom
160  PushVertex(RenderVertices,Temp,ThisLeft.X,ThisLeft.Y,UV,Colour,PriAtlas); // Right/Top
161  }
162  }

164  void LineList::_AppendVertices(ScreenVertexData& Vertices)
165  {
166  switch(Priority)

169  {
170  for( Whole X = 0 ; X < RenderVertices.size() ; ++X )
171  {
172  Vertices.LowVertices.push_back(RenderVertices[X]);

174  break;
175  }
176  case UI::RP_Medium:
