153  void UIManager::HotKeyAndInputCaptureChecks()
154  {
155  if(HotKeys.empty())
156  return;
157  std::vector<Input::InputCode> CapturedCodes;
158  std::list<EventUserInput*>* InputList = EventManager::GetSingletonPtr()->GetAllUserInputEvents();
159  Input::InputCode CurrCode;
160  for( std::list<EventUserInput*>::iterator Ilit = InputList->begin() ; Ilit != InputList->end() ; Ilit++ )
161  {
162  for( Whole X = 0 ; X < (*Ilit)->size() ; X++ )
163  {
164  CurrCode = (*Ilit)->GetMetaCode(X).GetCode();
165  if(LastWidgetSelected && LastWidgetSelected->IsInputCaptureWidget())
166  {
167  CapturedCodes.push_back(CurrCode);
168  }else{
169  HotKeyActivationCheck(CurrCode);
170  }
171  }
172  }
173  if(!CapturedCodes.empty())
174  LastWidgetSelected->GetInputCaptureData()->UpdateCapturedInputs(CapturedCodes);
175  }

177  void UIManager::WidgetUpdates()
178  {
179  if(HoveredWidget)
180  HoveredWidget->Update();
181  if(HoveredWidget || WidgetFocus)
182  {
183  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
184  if(Input::BUTTON_PRESSING == State)
185  {
186  WidgetFocus = HoveredWidget;
187  }
188  else if(Input::BUTTON_DOWN == State)
189  {
190  if(HoveredWidget != WidgetFocus && WidgetFocus)
191  WidgetFocus->Update(true);
192  }
193  else if(Input::BUTTON_LIFTING == State)
194  {
195  if(HoveredWidget != WidgetFocus && WidgetFocus)
196  WidgetFocus->Update(true);
197  else if(HoveredWidget && HoveredWidget == WidgetFocus)
198  LastWidgetSelected = HoveredWidget;
199  WidgetFocus = NULL;
200  }
201  }
202  else if(!HoveredWidget && !WidgetFocus)
203  {
204  Input::ButtonState State = Input::InputManager::GetSingletonPtr()->GetSystemMouse()->GetButtonState(1);
205  if(LastWidgetSelected && Input::BUTTON_LIFTING == State)
206  LastWidgetSelected = NULL;
207  }

225  void UIManager::MouseActivationCheck(Button* ToCheck)
226  {
227  if(!ToCheck)
228  return;
229 
230  Input::Mouse* SysMouse = Input::InputManager::GetSingletonPtr()->GetSystemMouse();
231  const std::vector<Input::InputCode>& MouseCodes = ToCheck->GetMouseActivationButtons();
232  Input::ButtonState State = ToCheck->IsLiftActivation() ? Input::BUTTON_LIFTING : Input::BUTTON_PRESSING;
233 
234  for( Whole X = 0 ; X < MouseCodes.size() ; X++ )
235  {
236  Input::InputCode Code = MouseCodes.at(X);
237  if( SysMouse->CheckButtonState(Code,State) )
238  {
239  ToCheck->_SetActivation(true);
240  ActivatedButtons.push_back(ToCheck);
241  }
242  }

478  UIManager::GlyphIndexResult UIManager::SuggestGlyphIndex(const Whole& Height, const String& Atlas)
479  {
480  TextureAtlas* TheAtlas = NULL;
481  AtlasIterator It = Atlases.find(Atlas);
482  if(It != Atlases.end()) TheAtlas = (*It).second;
483  else
484  {
485  MEZZ_EXCEPTION(Exception::PARAMETERS_EXCEPTION,"Attempting to access TextureAtlas \"" + Atlas + "\", which does not exist or is not loaded.");
486  }
487  std::map<UInt32,GlyphData*>& Glyphs = TheAtlas->GetGlyphs();
488  Whole LargerMatch = 0;
489  Whole SmallerMatch = 0;
490  Real LargerHeight = 0;
491  Real SmallerHeight = 0;
492  Real LargerMatchDiff = 1000000.0;
493  Real SmallerMatchDiff = 1000000.0;
494  Real RequestedHeight = (Real)Height;
495 
496  for( std::map<UInt32,GlyphData*>::iterator it = Glyphs.begin() ; it != Glyphs.end() ; it++ )
497  {
498  Real Diff = 0.0;
499  if((*it).second->LineHeight > RequestedHeight)
500  {
501  Diff = (*it).second->LineHeight - RequestedHeight;
502  if(Diff < LargerMatchDiff)
503  {
504  LargerMatch = (*it).first;
505  LargerHeight = (*it).second->LineHeight;
506  LargerMatchDiff = Diff;
507  continue;
508  }
509  }else{
510  Diff = RequestedHeight - (*it).second->LineHeight;
511  if(Diff < SmallerMatchDiff)
512  {
513  SmallerMatch = (*it).first;
514  SmallerHeight = (*it).second->LineHeight;
515  SmallerMatchDiff = Diff;
516  continue;
517  }
518  }
519  }
520 
521  Real Scale = 1;
522  if(LargerMatch != 0)
523  {
524  Scale = RequestedHeight / LargerHeight;
525  return GlyphIndexResult(LargerMatch,Scale);
526  }else{
527  Scale = RequestedHeight / SmallerHeight;
528  return GlyphIndexResult(SmallerMatch,Scale);
529  }
530  }
