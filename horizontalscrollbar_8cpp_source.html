209  bool HorizontalScrollbar::MouseScroll(const Vector2& MouseDelta)
210  {
211  if( this->ChildLock == this->Scroller && MouseDelta.X != 0 )
212  {
213  const Real OldScrollerValue = this->GetScrollerValue();
214  const Rect OldScrollerRect = this->Scroller->GetRect();
215  Rect NewScrollerRect;
216  NewScrollerRect.Size = OldScrollerRect.Size;
217 
218  NewScrollerRect.Position.X = OldScrollerRect.Position.X + MouseDelta.X;
219  NewScrollerRect.Position.Y = OldScrollerRect.Position.Y;
220 
221  NewScrollerRect.Position.X = std::min(NewScrollerRect.Position.X,this->GetUpperScrollLimit() - OldScrollerRect.Size.X);
222  NewScrollerRect.Position.X = std::max(NewScrollerRect.Position.X,this->GetLowerScrollLimit());
223 
224  this->Scroller->UpdateDimensions(OldScrollerRect,NewScrollerRect);
225 
226  const Real NewScrollerValue = this->GetScrollerValue();
227  this->_OnScrollValueChanged(OldScrollerValue,NewScrollerValue);
228 
229  return true;
230  }
231  return false;

234  bool HorizontalScrollbar::ScrollBackScroll(const Vector2& HitPosition)
235  {
236  if( this->ChildLock == this->ScrollBack )
237  {
238  const Real OldScrollerValue = this->GetScrollerValue();
239  const Rect OldScrollerRect = this->Scroller->GetRect();
240  Rect NewScrollerRect;
241  NewScrollerRect.Size = OldScrollerRect.Size;
242 
243  // Early escape in case of error, but in general this shouldn't happen.
244  if( HitPosition.X == -1.0 || HitPosition.Y == -1.0 )
245  return false;
246 
247  if( HitPosition.X < OldScrollerRect.Position.X ) { // Check for above
248  NewScrollerRect.Position.X = ( OldScrollerRect.Position.X - OldScrollerRect.Size.X >= this->GetLowerScrollLimit() ?
249  this->GetLowerScrollLimit() : OldScrollerRect.Position.X - OldScrollerRect.Size.X );
250  NewScrollerRect.Position.Y = OldScrollerRect.Position.Y;
251  }else if( HitPosition.X > OldScrollerRect.Position.X + OldScrollerRect.Size.X ) { // Check for below
252  NewScrollerRect.Position.X = ( OldScrollerRect.Position.X + ( OldScrollerRect.Size.X * 2 ) >= this->GetUpperScrollLimit() ?
253  this->GetUpperScrollLimit() - OldScrollerRect.Size.X : OldScrollerRect.Position.X + OldScrollerRect.Size.X );
254  NewScrollerRect.Position.Y = OldScrollerRect.Position.Y;
255  }
256 
257  this->Scroller->UpdateDimensions(OldScrollerRect,NewScrollerRect);
258 
259  const Real NewScrollerValue = this->GetScrollerValue();
260  this->_OnScrollValueChanged(OldScrollerValue,NewScrollerValue);
261 
262  return true;
263  }
264  return false;
265  }

319  void HorizontalScrollbar::SetScrollerValue(const Real& Value)
320  {
321  if(Value > 1 || Value < 0)
322  return;
323 
324  const Rect ScrollBackRect = this->ScrollBack->GetRect();
325  const Rect OldScrollerRect = this->Scroller->GetRect();
326  Rect NewScrollerRect;
327 
328  NewScrollerRect.Size.X = OldScrollerRect.Size.X;
329  NewScrollerRect.Size.Y = OldScrollerRect.Size.Y;
330  NewScrollerRect.Position.X = ScrollBackRect.Position.X + ( ( this->GetScrollRange() - NewScrollerRect.Size.X ) * Value );
331  NewScrollerRect.Position.Y = ScrollBackRect.Position.Y;
332 
333  this->Scroller->UpdateDimensions(OldScrollerRect,NewScrollerRect);
334  }

336  Real HorizontalScrollbar::GetScrollerValue() const

343  void HorizontalScrollbar::SetScrollerSize(const Real& Size)
344  {
345  if(Size > 1 || Size < 0)
346  return;
347 
348  const Rect ScrollBackRect = this->ScrollBack->GetRect();
349  const Rect OldScrollerRect = this->Scroller->GetRect();
350  Rect NewScrollerRect;
351 
352  Real ScrollValue = this->GetScrollerValue();
353  NewScrollerRect.Size.X = ScrollBackRect.Size.X * Size;
354  NewScrollerRect.Size.Y = ScrollBackRect.Size.Y;
355  NewScrollerRect.Position.X = ScrollBackRect.Position.X + ( ( this->GetScrollRange() - NewScrollerRect.Size.X ) * ScrollValue );
356  NewScrollerRect.Position.Y = ScrollBackRect.Position.Y;
357 
358  this->Scroller->UpdateDimensions(OldScrollerRect,NewScrollerRect);
359  }

366  void HorizontalScrollbar::UpdateDimensions(const Rect& OldSelfRect, const Rect& NewSelfRect)
367  {
368  // Update the personal data first
369  this->ActDims = NewSelfRect;
370 
371  // Get the data regarding the scroller we'll need after everything else is updated
372  Real ScrollValue = this->GetScrollerValue();
373  Real ScrollSize = this->GetScrollerSize();
374  const Rect OldScrollerRect = this->Scroller->GetRect();
375 
376  // Update the children
377  this->LayoutStrat->Layout(OldSelfRect,NewSelfRect,this->ChildWidgets);
378 
379  // Next prepare the new rect for the scroller
380  const Rect ScrollBackRect = this->ScrollBack->GetRect();
381  Rect NewScrollerRect;
382  NewScrollerRect.Size.X = ScrollBackRect.Size.X * ScrollSize;
383  NewScrollerRect.Size.Y = ScrollBackRect.Size.Y;
384  NewScrollerRect.Position.X = ScrollBackRect.Position.X + ( ( this->GetScrollRange() - NewScrollerRect.Size.Y ) * ScrollValue );
385  NewScrollerRect.Position.Y = ScrollBackRect.Position.Y;
386 
387  // Finally update the scroller
388  this->Scroller->UpdateDimensions(OldScrollerRect,NewScrollerRect);
389 
390  // We done got icky
391  this->_MarkAllLayersDirty();
392  }
